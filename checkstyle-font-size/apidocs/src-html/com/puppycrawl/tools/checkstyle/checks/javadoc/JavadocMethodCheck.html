046/**
047 * <p>
048 * Checks the Javadoc of a method or constructor.
049 * The scope to verify is specified using the {@code Scope} class and defaults
050 * to {@code Scope.PRIVATE}. To verify another scope, set property scope to
051 * a different <a href="https://checkstyle.org/property_types.html#scope">scope</a>.
052 * </p>
053 * <p>
054 * Violates parameters and type parameters for which no param tags are present can
055 * be suppressed by defining property {@code allowMissingParamTags}.
056 * </p>
057 * <p>
058 * Violates methods which return non-void but for which no return tag is present can
059 * be suppressed by defining property {@code allowMissingReturnTag}.
060 * </p>
061 * <p>
062 * Violates exceptions which are declared to be thrown (by {@code throws} in the method
063 * signature or by {@code throw new} in the method body), but for which no throws tag is
064 * present by activation of property {@code validateThrows}.
065 * Note that {@code throw new} is not checked in the following places:
066 * </p>
067 * <ul>
068 * <li>
069 * Inside a try block (with catch). It is not possible to determine if the thrown
070 * exception can be caught by the catch block as there is no knowledge of the
071 * inheritance hierarchy, so the try block is ignored entirely. However, catch
072 * and finally blocks, as well as try blocks without catch, are still checked.
073 * </li>
074 * <li>
075 * Local classes, anonymous classes and lambda expressions. It is not known when the
076 * throw statements inside such classes are going to be evaluated, so they are ignored.
077 * </li>
078 * </ul>
079 * <p>
080 * ATTENTION: Checkstyle does not have information about hierarchy of exception types
081 * so usage of base class is considered as separate exception type.
082 * As workaround you need to specify both types in javadoc (parent and exact type).
083 * </p>
084 * <p>
085 * Javadoc is not required on a method that is tagged with the {@code @Override}
086 * annotation. However under Java 5 it is not possible to mark a method required
087 * for an interface (this was <i>corrected</i> under Java 6). Hence Checkstyle
088 * supports using the convention of using a single {@code {@inheritDoc}} tag
089 * instead of all the other tags.
090 * </p>
091 * <p>
092 * Note that only inheritable items will allow the {@code {@inheritDoc}}
093 * tag to be used in place of comments. Static methods at all visibilities,
094 * private non-static methods and constructors are not inheritable.
095 * </p>
096 * <p>
097 * For example, if the following method is implementing a method required by
098 * an interface, then the Javadoc could be done as:
099 * </p>
100 * <pre>
101 * &#47;** {&#64;inheritDoc} *&#47;
102 * public int checkReturnTag(final int aTagIndex,
103 *                           JavadocTag[] aTags,
104 *                           int aLineNo)
105 * </pre>
106 * <ul>
107 * <li>
108 * Property {@code allowedAnnotations} - Specify the list of annotations
109 * that allow missed documentation.
110 * Default value is {@code Override}.
111 * </li>
112 * <li>
113 * Property {@code validateThrows} - Control whether to validate {@code throws} tags.
114 * Default value is {@code false}.
115 * </li>
116 * <li>
117 * Property {@code scope} - Specify the visibility scope where Javadoc comments are checked.
118 * Default value is {@code private}.
119 * </li>
120 * <li>
121 * Property {@code excludeScope} - Specify the visibility scope where Javadoc comments
122 * are not checked.
123 * Default value is {@code null}.
124 * </li>
125 * <li>
126 * Property {@code allowMissingParamTags} - Control whether to ignore violations
127 * when a method has parameters but does not have matching {@code param} tags in the javadoc.
128 * Default value is {@code false}.
129 * </li>
130 * <li>
131 * Property {@code allowMissingReturnTag} - Control whether to ignore violations
132 * when a method returns non-void type and does not have a {@code return} tag in the javadoc.
133 * Default value is {@code false}.
134 * </li>
135 * <li>
136 * Property {@code tokens} - tokens to check Default value is:
137 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
138 * METHOD_DEF</a>,
139 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CTOR_DEF">
140 * CTOR_DEF</a>,
141 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_FIELD_DEF">
142 * ANNOTATION_FIELD_DEF</a>.
143 * </li>
144 * </ul>
145 * <p>
146 * To configure the default check:
147 * </p>
148 * <pre>
149 * &lt;module name="JavadocMethod"/&gt;
150 * </pre>
151 * <p>
152 * To configure the check for {@code public} scope, ignoring any missing param tags is:
153 * </p>
154 * <pre>
155 * &lt;module name="JavadocMethod"&gt;
156 *   &lt;property name="scope" value="public"/&gt;
157 *   &lt;property name="allowMissingParamTags" value="true"/&gt;
158 * &lt;/module&gt;
159 * </pre>
160 * <p>
161 * To configure the check for methods which are in {@code private},
162 * but not in {@code protected} scope:
163 * </p>
164 * <pre>
165 * &lt;module name="JavadocMethod"&gt;
166 *   &lt;property name="scope" value="private"/&gt;
167 *   &lt;property name="excludeScope" value="protected"/&gt;
168 * &lt;/module&gt;
169 * </pre>
170 * <p>
171 * To configure the check to validate {@code throws} tags, you can use following config.
172 * </p>
173 * <pre>
174 * &lt;module name="JavadocMethod"&gt;
175 *   &lt;property name="validateThrows" value="true"/&gt;
176 * &lt;/module&gt;
177 * </pre>
178 * <pre>
179 * &#47;**
180 *  * Actual exception thrown is child class of class that is declared in throws.
181 *  * It is limitation of checkstyle (as checkstyle does not know type hierarchy).
182 *  * Javadoc is valid not declaring FileNotFoundException
183 *  * BUT checkstyle can not distinguish relationship between exceptions.
184 *  * &#64;param file some file
185 *  * &#64;throws IOException if some problem
186 *  *&#47;
187 * public void doSomething8(File file) throws IOException {
188 *     if (file == null) {
189 *         throw new FileNotFoundException(); // violation
190 *     }
191 * }
192 *
193 * &#47;**
194 *  * Exact throw type referencing in javadoc even first is parent of second type.
195 *  * It is a limitation of checkstyle (as checkstyle does not know type hierarchy).
196 *  * This javadoc is valid for checkstyle and for javadoc tool.
197 *  * &#64;param file some file
198 *  * &#64;throws IOException if some problem
199 *  * &#64;throws FileNotFoundException if file is not found
200 *  *&#47;
201 * public void doSomething9(File file) throws IOException {
202 *     if (file == null) {
203 *         throw new FileNotFoundException();
204 *     }
205 * }
206 *
207 * &#47;**
208 *  * Ignore try block, but keep catch and finally blocks.
209 *  *
210 *  * &#64;param s String to parse
211 *  * &#64;return A positive integer
212 *  *&#47;
213 * public int parsePositiveInt(String s) {
214 *     try {
215 *         int value = Integer.parseInt(s);
216 *         if (value &lt;= 0) {
217 *             throw new NumberFormatException(value + " is negative/zero"); // ok, try
218 *         }
219 *         return value;
220 *     } catch (NumberFormatException ex) {
221 *         throw new IllegalArgumentException("Invalid number", ex); // violation, catch
222 *     } finally {
223 *         throw new IllegalStateException("Should never reach here"); // violation, finally
224 *     }
225 * }
226 *
227 * &#47;**
228 *  * Try block without catch is not ignored.
229 *  *
230 *  * &#64;return a String from standard input, if there is one
231 *  *&#47;
232 * public String readLine() {
233 *     try (Scanner sc = new Scanner(System.in)) {
234 *         if (!sc.hasNext()) {
235 *             throw new IllegalStateException("Empty input"); // violation, not caught
236 *         }
237 *         return sc.next();
238 *     }
239 * }
240 *
241 * &#47;**
242 *  * Lambda expressions are ignored as we do not know when the exception will be thrown.
243 *  *
244 *  * &#64;param s a String to be printed at some point in the future
245 *  * &#64;return a Runnable to be executed when the string is to be printed
246 *  *&#47;
247 * public Runnable printLater(String s) {
248 *     return () -&gt; {
249 *         if (s == null) {
250 *             throw new NullPointerException(); // ok
251 *         }
252 *         System.out.println(s);
253 *     };
254 * }
255 * </pre>
256 *
257 * @since 3.0
258 */
259@FileStatefulCheck

470    /**
471     * Called to process an AST when visiting it.
472     *
473     * @param ast the AST to process. Guaranteed to not be PACKAGE_DEF or
474     *             IMPORT tokens.
475     */
476    private void processAST(DetailAST ast) {
477        final Scope theScope = calculateScope(ast);
478        if (shouldCheck(ast, theScope)) {
479            final FileContents contents = getFileContents();
480            final TextBlock textBlock = contents.getJavadocBefore(ast.getLineNo());
481
482            if (textBlock != null) {
483                checkComment(ast, textBlock);
484            }
485        }
486    }

488    /**
489     * Whether we should check this node.
490     *
491     * @param ast a given node.
492     * @param nodeScope the scope of the node.
493     * @return whether we should check a given node.
494     */
495    private boolean shouldCheck(final DetailAST ast, final Scope nodeScope) {
496        final Scope surroundingScope = ScopeUtil.getSurroundingScope(ast);
497
498        return (excludeScope == null
499                || nodeScope != excludeScope
500                && surroundingScope != excludeScope)
501            && nodeScope.isIn(scope)
502            && surroundingScope.isIn(scope);
503    }

505    /**
506     * Checks the Javadoc for a method.
507     *
508     * @param ast the token for the method
509     * @param comment the Javadoc comment
510     */
511    private void checkComment(DetailAST ast, TextBlock comment) {
512        final List<JavadocTag> tags = getMethodTags(comment);
513
514        if (!hasShortCircuitTag(ast, tags)) {
515            if (ast.getType() == TokenTypes.ANNOTATION_FIELD_DEF) {
516                checkReturnTag(tags, ast.getLineNo(), true);
517            }
518            else {
519                final Iterator<JavadocTag> it = tags.iterator();
520                // Check for inheritDoc
521                boolean hasInheritDocTag = false;
522                while (!hasInheritDocTag && it.hasNext()) {
523                    hasInheritDocTag = it.next().isInheritDocTag();
524                }
525                final boolean reportExpectedTags = !hasInheritDocTag
526                    && !AnnotationUtil.containsAnnotation(ast, allowedAnnotations);
527
528                checkParamTags(tags, ast, reportExpectedTags);
529                final List<ExceptionInfo> throwed =
530                        combineExceptionInfo(getThrows(ast), getThrowed(ast));
531                checkThrowsTags(tags, throwed, reportExpectedTags);
532                if (CheckUtil.isNonVoidMethod(ast)) {
533                    checkReturnTag(tags, ast.getLineNo(), reportExpectedTags);
534                }
535            }
536
537            // Dump out all unused tags
538            tags.stream().filter(javadocTag -> !javadocTag.isSeeOrInheritDocTag())
539                .forEach(javadocTag -> log(javadocTag.getLineNo(), MSG_UNUSED_TAG_GENERAL));
540        }
541    }

543    /**
544     * Validates whether the Javadoc has a short circuit tag. Currently this is
545     * the inheritTag. Any violations are logged.
546     *
547     * @param ast the construct being checked
548     * @param tags the list of Javadoc tags associated with the construct
549     * @return true if the construct has a short circuit tag.
550     */
551    private boolean hasShortCircuitTag(final DetailAST ast, final List<JavadocTag> tags) {
552        boolean result = true;
553        // Check if it contains {@inheritDoc} tag
554        if (tags.size() == 1
555                && tags.get(0).isInheritDocTag()) {
556            // Invalid if private, a constructor, or a static method
557            if (!JavadocTagInfo.INHERIT_DOC.isValidOn(ast)) {
558                log(ast, MSG_INVALID_INHERIT_DOC);
559            }
560        }
561        else {
562            result = false;
563        }
564        return result;
565    }

567    /**
568     * Returns the scope for the method/constructor at the specified AST. If
569     * the method is in an interface or annotation block, the scope is assumed
570     * to be public.
571     *
572     * @param ast the token of the method/constructor
573     * @return the scope of the method/constructor
574     */
575    private static Scope calculateScope(final DetailAST ast) {
576        final Scope scope;
577
578        if (ScopeUtil.isInInterfaceOrAnnotationBlock(ast)) {
579            scope = Scope.PUBLIC;
580        }
581        else {
582            final DetailAST mods = ast.findFirstToken(TokenTypes.MODIFIERS);
583            scope = ScopeUtil.getScopeFromMods(mods);
584        }
585        return scope;
586    }

588    /**
589     * Returns the tags in a javadoc comment. Only finds throws, exception,
590     * param, return and see tags.
591     *
592     * @param comment the Javadoc comment
593     * @return the tags found
594     */
595    private static List<JavadocTag> getMethodTags(TextBlock comment) {
596        final String[] lines = comment.getText();
597        final List<JavadocTag> tags = new ArrayList<>();
598        int currentLine = comment.getStartLineNo() - 1;
599        final int startColumnNumber = comment.getStartColNo();
600
601        for (int i = 0; i < lines.length; i++) {
602            currentLine++;
603            final Matcher javadocArgMatcher =
604                MATCH_JAVADOC_ARG.matcher(lines[i]);
605            final Matcher javadocArgMissingDescriptionMatcher =
606                MATCH_JAVADOC_ARG_MISSING_DESCRIPTION.matcher(lines[i]);
607            final Matcher javadocNoargMatcher =
608                MATCH_JAVADOC_NOARG.matcher(lines[i]);
609            final Matcher noargCurlyMatcher =
610                MATCH_JAVADOC_NOARG_CURLY.matcher(lines[i]);
611            final Matcher noargMultilineStart =
612                MATCH_JAVADOC_NOARG_MULTILINE_START.matcher(lines[i]);
613
614            if (javadocArgMatcher.find()) {
615                final int col = calculateTagColumn(javadocArgMatcher, i, startColumnNumber);
616                tags.add(new JavadocTag(currentLine, col, javadocArgMatcher.group(1),
617                        javadocArgMatcher.group(2)));
618            }
619            else if (javadocArgMissingDescriptionMatcher.find()) {
620                final int col = calculateTagColumn(javadocArgMissingDescriptionMatcher, i,
621                    startColumnNumber);
622                tags.add(new JavadocTag(currentLine, col,
623                    javadocArgMissingDescriptionMatcher.group(1),
624                    javadocArgMissingDescriptionMatcher.group(2)));
625            }
626            else if (javadocNoargMatcher.find()) {
627                final int col = calculateTagColumn(javadocNoargMatcher, i, startColumnNumber);
628                tags.add(new JavadocTag(currentLine, col, javadocNoargMatcher.group(1)));
629            }
630            else if (noargCurlyMatcher.find()) {
631                final int col = calculateTagColumn(noargCurlyMatcher, i, startColumnNumber);
632                tags.add(new JavadocTag(currentLine, col, noargCurlyMatcher.group(1)));
633            }
634            else if (noargMultilineStart.find()) {
635                tags.addAll(getMultilineNoArgTags(noargMultilineStart, lines, i, currentLine));
636            }
637        }
638        return tags;
639    }

641    /**
642     * Calculates column number using Javadoc tag matcher.
643     *
644     * @param javadocTagMatcher found javadoc tag matcher
645     * @param lineNumber line number of Javadoc tag in comment
646     * @param startColumnNumber column number of Javadoc comment beginning
647     * @return column number
648     */
649    private static int calculateTagColumn(Matcher javadocTagMatcher,
650            int lineNumber, int startColumnNumber) {
651        int col = javadocTagMatcher.start(1) - 1;
652        if (lineNumber == 0) {
653            col += startColumnNumber;
654        }
655        return col;
656    }

658    /**
659     * Gets multiline Javadoc tags with no arguments.
660     *
661     * @param noargMultilineStart javadoc tag Matcher
662     * @param lines comment text lines
663     * @param lineIndex line number that contains the javadoc tag
664     * @param tagLine javadoc tag line number in file
665     * @return javadoc tags with no arguments
666     */
667    private static List<JavadocTag> getMultilineNoArgTags(final Matcher noargMultilineStart,
668            final String[] lines, final int lineIndex, final int tagLine) {
669        int remIndex = lineIndex;
670        Matcher multilineCont;
671
672        do {
673            remIndex++;
674            multilineCont = MATCH_JAVADOC_MULTILINE_CONT.matcher(lines[remIndex]);
675        } while (!multilineCont.find());
676
677        final List<JavadocTag> tags = new ArrayList<>();
678        final String lFin = multilineCont.group(1);
679        if (!lFin.equals(NEXT_TAG)
680            && !lFin.equals(END_JAVADOC)) {
681            final String param1 = noargMultilineStart.group(1);
682            final int col = noargMultilineStart.start(1) - 1;
683
684            tags.add(new JavadocTag(tagLine, col, param1));
685        }
686
687        return tags;
688    }

692     *

694     * @return the list of parameter nodes for ast.

696    private static List<DetailAST> getParameters(DetailAST ast) {

698        final List<DetailAST> returnValue = new ArrayList<>();
699

701        while (child != null) {

703                final DetailAST ident = child.findFirstToken(TokenTypes.IDENT);
704                if (ident != null) {
705                    returnValue.add(ident);
706                }
707            }
708            child = child.getNextSibling();
709        }
710        return returnValue;
711    }

737     * Get ExceptionInfo for all exceptions that throws in method code by 'throw new'.
738     *
739     * @param methodAst method DetailAST object where to find exceptions
740     * @return list of ExceptionInfo
741     */
742    private static List<ExceptionInfo> getThrowed(DetailAST methodAst) {
743        final List<ExceptionInfo> returnValue = new ArrayList<>();
744        final DetailAST blockAst = methodAst.findFirstToken(TokenTypes.SLIST);
745        if (blockAst != null) {
746            final List<DetailAST> throwLiterals = findTokensInAstByType(blockAst,
747                    TokenTypes.LITERAL_THROW);
748            for (DetailAST throwAst : throwLiterals) {
749                if (!isInIgnoreBlock(blockAst, throwAst)) {
750                    final DetailAST newAst = throwAst.getFirstChild().getFirstChild();
751                    if (newAst.getType() == TokenTypes.LITERAL_NEW) {
752                        final DetailAST child = newAst.getFirstChild();
753                        returnValue.add(getExceptionInfo(child));
754                    }
755                }
756            }
757        }
758        return returnValue;
759    }

788    /**
789     * Checks if a 'throw' usage is contained within a block that should be ignored.
790     * Such blocks consist of try (with catch) blocks, local classes, anonymous classes,
791     * and lambda expressions. Note that a try block without catch is not considered.
792     *
793     * @param methodBodyAst DetailAST node representing the method body
794     * @param throwAst DetailAST node representing the 'throw' literal
795     * @return true if throwAst is inside a block that should be ignored
796     */
797    private static boolean isInIgnoreBlock(DetailAST methodBodyAst, DetailAST throwAst) {
798        DetailAST ancestor = throwAst.getParent();
799        while (ancestor != methodBodyAst) {
800            if (ancestor.getType() == TokenTypes.LITERAL_TRY
801                    && ancestor.findFirstToken(TokenTypes.LITERAL_CATCH) != null
802                    || ancestor.getType() == TokenTypes.LAMBDA
803                    || ancestor.getType() == TokenTypes.OBJBLOCK) {
804                // throw is inside a try block, and there is a catch block,
805                // or throw is inside a lambda expression/anonymous class/local class
806                break;
807            }
808            if (ancestor.getType() == TokenTypes.LITERAL_CATCH
809                    || ancestor.getType() == TokenTypes.LITERAL_FINALLY) {
810                // if the throw is inside a catch or finally block,
811                // skip the immediate ancestor (try token)
812                ancestor = ancestor.getParent();
813            }
814            ancestor = ancestor.getParent();
815        }
816        return ancestor != methodBodyAst;
817    }

821     *

823     * @param list2 list of ExceptionInfo
824     * @return combined list of ExceptionInfo

826    private static List<ExceptionInfo> combineExceptionInfo(List<ExceptionInfo> list1,

829        for (ExceptionInfo exceptionInfo : list2) {

831                result.add(exceptionInfo);

837    /**
838     * Finds node of specified type among root children, siblings, siblings children
839     * on any deep level.
840     *
841     * @param root    DetailAST
842     * @param astType value of TokenType
843     * @return {@link List} of {@link DetailAST} nodes which matches the predicate.
844     */
845    public static List<DetailAST> findTokensInAstByType(DetailAST root, int astType) {
846        final List<DetailAST> result = new ArrayList<>();
847        DetailAST curNode = root;
848        while (curNode != null) {
849            DetailAST toVisit = curNode.getFirstChild();
850            while (curNode != null && toVisit == null) {
851                toVisit = curNode.getNextSibling();
852                curNode = curNode.getParent();
853                if (curNode == root) {
854                    toVisit = null;
855                    break;
856                }
857            }
858            curNode = toVisit;
859            if (curNode != null && curNode.getType() == astType) {
860                result.add(curNode);
861            }
862        }
863        return result;
864    }

866    /**
867     * Checks a set of tags for matching parameters.
868     *
869     * @param tags the tags to check
870     * @param parent the node which takes the parameters
871     * @param reportExpectedTags whether we should report if do not find
872     *            expected tag
873     */
874    private void checkParamTags(final List<JavadocTag> tags,
875            final DetailAST parent, boolean reportExpectedTags) {
876        final List<DetailAST> params = getParameters(parent);
877        final List<DetailAST> typeParams = CheckUtil
878                .getTypeParameters(parent);
879
880        // Loop over the tags, checking to see they exist in the params.
881        final ListIterator<JavadocTag> tagIt = tags.listIterator();
882        while (tagIt.hasNext()) {
883            final JavadocTag tag = tagIt.next();
884
885            if (!tag.isParamTag()) {
886                continue;
887            }
888
889            tagIt.remove();
890
891            final String arg1 = tag.getFirstArg();
892            boolean found = removeMatchingParam(params, arg1);
893
894            if (CommonUtil.startsWithChar(arg1, '<') && CommonUtil.endsWithChar(arg1, '>')) {
895                found = searchMatchingTypeParameter(typeParams,
896                        arg1.substring(1, arg1.length() - 1));
897            }
898
899            // Handle extra JavadocTag
900            if (!found) {
901                log(tag.getLineNo(), tag.getColumnNo(), MSG_UNUSED_TAG,
902                        "@param", arg1);
903            }
904        }
905
906        // Now dump out all type parameters/parameters without tags :- unless
907        // the user has chosen to suppress these problems
908        if (!allowMissingParamTags && reportExpectedTags) {
909            for (DetailAST param : params) {
910                log(param, MSG_EXPECTED_TAG,
911                    JavadocTagInfo.PARAM.getText(), param.getText());
912            }
913
914            for (DetailAST typeParam : typeParams) {
915                log(typeParam, MSG_EXPECTED_TAG,
916                    JavadocTagInfo.PARAM.getText(),
917                    "<" + typeParam.findFirstToken(TokenTypes.IDENT).getText()
918                    + ">");
919            }
920        }
921    }

924     * Returns true if required type found in type parameters.
925     *
926     * @param typeParams
927     *            list of type parameters
928     * @param requiredTypeName
929     *            name of required type
930     * @return true if required type found in type parameters.
931     */
932    private static boolean searchMatchingTypeParameter(List<DetailAST> typeParams,
933            String requiredTypeName) {
934        // Loop looking for matching type param
935        final Iterator<DetailAST> typeParamsIt = typeParams.iterator();
936        boolean found = false;
937        while (typeParamsIt.hasNext()) {
938            final DetailAST typeParam = typeParamsIt.next();
939            if (typeParam.findFirstToken(TokenTypes.IDENT).getText()
940                    .equals(requiredTypeName)) {
941                found = true;
942                typeParamsIt.remove();
943                break;
944            }
945        }
946        return found;
947    }

971     * Checks for only one return tag. All return tags will be removed from the
972     * supplied list.
973     *
974     * @param tags the tags to check
975     * @param lineNo the line number of the expected tag
976     * @param reportExpectedTags whether we should report if do not find
977     *            expected tag
978     */
979    private void checkReturnTag(List<JavadocTag> tags, int lineNo,
980        boolean reportExpectedTags) {
981        // Loop over tags finding return tags. After the first one, report an
982        // violation.
983        boolean found = false;
984        final ListIterator<JavadocTag> it = tags.listIterator();
985        while (it.hasNext()) {
986            final JavadocTag javadocTag = it.next();
987            if (javadocTag.isReturnTag()) {
988                if (found) {
989                    log(javadocTag.getLineNo(), javadocTag.getColumnNo(),
990                            MSG_DUPLICATE_TAG,
991                            JavadocTagInfo.RETURN.getText());
992                }
993                found = true;
994                it.remove();
995            }
996        }
997
998        // Handle there being no @return tags :- unless
999        // the user has chosen to suppress these problems
1000        if (!found && !allowMissingReturnTag && reportExpectedTags) {
1001            log(lineNo, MSG_RETURN_EXPECTED);
1002        }
1003    }

1005    /**
1006     * Checks a set of tags for matching throws.
1007     *
1008     * @param tags the tags to check
1009     * @param throwsList the throws to check
1010     * @param reportExpectedTags whether we should report if do not find
1011     *            expected tag
1012     */
1013    private void checkThrowsTags(List<JavadocTag> tags,
1014            List<ExceptionInfo> throwsList, boolean reportExpectedTags) {
1015        // Loop over the tags, checking to see they exist in the throws.
1016        // The foundThrows used for performance only
1017        final Set<String> foundThrows = new HashSet<>();
1018        final ListIterator<JavadocTag> tagIt = tags.listIterator();
1019        while (tagIt.hasNext()) {
1020            final JavadocTag tag = tagIt.next();
1021
1022            if (!tag.isThrowsTag()) {
1023                continue;
1024            }
1025            tagIt.remove();
1026
1027            // Loop looking for matching throw
1028            final Token token = new Token(tag.getFirstArg(), tag.getLineNo(), tag
1029                    .getColumnNo());
1030            final ClassInfo documentedClassInfo = new ClassInfo(token);
1031            processThrows(throwsList, documentedClassInfo, foundThrows);
1032        }
1033        // Now dump out all throws without tags :- unless
1034        // the user has chosen to suppress these problems
1035        if (validateThrows && reportExpectedTags) {
1036            throwsList.stream().filter(exceptionInfo -> !exceptionInfo.isFound())
1037                .forEach(exceptionInfo -> {
1038                    final Token token = exceptionInfo.getName();
1039                    log(exceptionInfo.getAst(),
1040                        MSG_EXPECTED_TAG,
1041                        JavadocTagInfo.THROWS.getText(), token.getText());
1042                });
1043        }
1044    }

1046    /**
1047     * Verifies that documented exception is in throws.
1048     *
1049     * @param throwsList list of throws
1050     * @param documentedClassInfo documented exception class info
1051     * @param foundThrows previously found throws
1052     */
1053    private static void processThrows(List<ExceptionInfo> throwsList,
1054                                      ClassInfo documentedClassInfo, Set<String> foundThrows) {
1055        ExceptionInfo foundException = null;
1056
1057        // First look for matches on the exception name
1058        for (ExceptionInfo exceptionInfo : throwsList) {
1059            if (isClassNamesSame(exceptionInfo.getName().getText(),
1060                    documentedClassInfo.getName().getText())) {
1061                foundException = exceptionInfo;
1062                break;
1063            }
1064        }
1065
1066        if (foundException != null) {
1067            foundException.setFound();
1068            foundThrows.add(documentedClassInfo.getName().getText());
1069        }
1070    }

1084    /**
1085     * Check that class names are same by short name of class. If some class name is fully
1086     * qualified it is cut to short name.
1087     *
1088     * @param class1 class name
1089     * @param class2 class name
1090     * @return true is ExceptionInfo object have the same name
1091     */
1092    private static boolean isClassNamesSame(String class1, String class2) {
1093        boolean result = false;
1094        if (class1.equals(class2)) {
1095            result = true;
1096        }
1097        else {
1098            final String separator = ".";
1099            if (class1.contains(separator) || class2.contains(separator)) {
1100                final String class1ShortName = class1
1101                        .substring(class1.lastIndexOf('.') + 1);
1102                final String class2ShortName = class2
1103                        .substring(class2.lastIndexOf('.') + 1);
1104                result = class1ShortName.equals(class2ShortName);
1105            }
1106        }
1107        return result;
1108    }
