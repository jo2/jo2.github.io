033/**
034 * <p>
035 * Checks the ordering/grouping of imports. Features are:
036 * </p>
037 * <ul>
038 * <li>
039 * groups type/static imports: ensures that groups of imports come in a specific order
040 * (e.g., java. comes first, javax. comes second, then everything else)
041 * </li>
042 * <li>
043 * adds a separation between type import groups : ensures that a blank line sit between each group
044 * </li>
045 * <li>
046 * type/static import groups aren't separated internally: ensures that each group aren't separated
047 * internally by blank line or comment
048 * </li>
049 * <li>
050 * sorts type/static imports inside each group: ensures that imports within each group are in
051 * lexicographic order
052 * </li>
053 * <li>
054 * sorts according to case: ensures that the comparison between imports is case sensitive, in
055 * <a href="https://en.wikipedia.org/wiki/ASCII#Order">ASCII sort order</a>
056 * </li>
057 * <li>
058 * arrange static imports: ensures the relative order between type imports and static imports
059 * (see
060 * <a href="https://checkstyle.org/property_types.html#ImportOrderOption">ImportOrderOption</a>)
061 * </li>
062 * </ul>
063 * <ul>
064 * <li>
065 * Property {@code option} - specify policy on the relative order between type imports and static
066 * imports.
067 * Default value is {@code under}.
068 * </li>
069 * <li>
070 * Property {@code groups} - specify list of <b>type import</b> groups (every group identified
071 * either by a common prefix string, or by a regular expression enclosed in forward slashes
072 * (e.g. {@code /regexp/}). All type imports, which does not match any group, falls into an
073 * additional group, located at the end.
074 * Thus, the empty list of type groups (the default value) means one group for all type imports.
075 * Default value is {@code {}}.
076 * </li>
077 * <li>
078 * Property {@code ordered} - control whether type imports within each group should be
079 * sorted.
080 * It doesn't affect sorting for static imports.
081 * Default value is true.
082 * </li>
083 * <li>
084 * Property {@code separated} - control whether type import groups should be separated
085 * by, at least, one blank line or comment and aren't separated internally.
086 * It doesn't affect separations for static imports.
087 * Default value is false.
088 * </li>
089 * <li>
090 * Property {@code separatedStaticGroups} - control whether static import groups should
091 * be separated by, at least, one blank line or comment and aren't separated internally.
092 * This property has effect only when the property {@code option} is is set to {@code top}
093 * or {@code bottom}.
094 * Default value is false.
095 * </li>
096 * <li>
097 * Property {@code caseSensitive} - control whether string comparison should be case
098 * sensitive or not. Case sensitive sorting is in
099 * <a href="https://en.wikipedia.org/wiki/ASCII#Order">ASCII sort order</a>.
100 * It affects both type imports and static imports.
101 * Default value is true.
102 * </li>
103 * <li>
104 * Property {@code staticGroups} - specify list of <b>static</b> import groups (every group
105 * identified either by a common prefix string, or by a regular expression enclosed in forward
106 * slashes (e.g. {@code /regexp/}). All static imports, which does not match any group, falls into
107 * an additional group, located at the end. Thus, the empty list of static groups (the default
108 * value) means one group for all static imports. This property has effect only when the property
109 * {@code option} is set to {@code top} or {@code bottom}.
110 * Default value is {@code {}}.
111 * </li>
112 * <li>
113 * Property {@code sortStaticImportsAlphabetically} - control whether
114 * <b>static imports</b> located at <b>top</b> or <b>bottom</b> are sorted within the group.
115 * Default value is false.
116 * </li>
117 * <li>
118 * Property {@code useContainerOrderingForStatic} - control whether to use container
119 * ordering (Eclipse IDE term) for static imports or not.
120 * Default value is false.
121 * </li>
122 * <li>
123 * Property {@code tokens} - tokens to check
124 * Default value is:
125 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#STATIC_IMPORT">
126 * STATIC_IMPORT</a>.
127 * </li>
128 * </ul>
129 * <p>
130 * To configure the check so that it matches default Eclipse formatter configuration
131 * (tested on Kepler and Luna releases):
132 * </p>
133 * <ul>
134 * <li>
135 * group of static imports is on the top
136 * </li>
137 * <li>
138 * groups of type imports: "java" and "javax" packages first, then "org" and then all other imports
139 * </li>
140 * <li>
141 * imports will be sorted in the groups
142 * </li>
143 * <li>
144 * groups are separated by, at least, one blank line and aren't separated internally
145 * </li>
146 * </ul>
147 * <p>
148 * Notes:
149 * </p>
150 * <ul>
151 * <li>
152 * "com" package is not mentioned on configuration, because it is ignored by Eclipse Kepler and Luna
153 * (looks like Eclipse defect)
154 * </li>
155 * <li>
156 * configuration below doesn't work in all 100% cases due to inconsistent behavior prior to
157 * Mars release, but covers most scenarios
158 * </li>
159 * </ul>
160 * <pre>
161 * &lt;module name=&quot;ImportOrder&quot;&gt;
162 *   &lt;property name=&quot;groups&quot; value=&quot;/^java\./,javax,org&quot;/&gt;
163 *   &lt;property name=&quot;ordered&quot; value=&quot;true&quot;/&gt;
164 *   &lt;property name=&quot;separated&quot; value=&quot;true&quot;/&gt;
165 *   &lt;property name=&quot;option&quot; value=&quot;above&quot;/&gt;
166 *   &lt;property name=&quot;sortStaticImportsAlphabetically&quot; value=&quot;true&quot;/&gt;
167 * &lt;/module&gt;
168 * </pre>
169 * <p>
170 * To configure the check so that it matches default Eclipse formatter configuration
171 * (tested on Mars release):
172 * </p>
173 * <ul>
174 * <li>
175 * group of static imports is on the top
176 * </li>
177 * <li>
178 * groups of type imports: "java" and "javax" packages first, then "org" and "com",
179 * then all other imports as one group
180 * </li>
181 * <li>
182 * imports will be sorted in the groups
183 * </li>
184 * <li>
185 * groups are separated by, at least, one blank line and aren't separated internally
186 * </li>
187 * </ul>
188 * <pre>
189 * &lt;module name=&quot;ImportOrder&quot;&gt;
190 *   &lt;property name=&quot;groups&quot; value=&quot;/^java\./,javax,org,com&quot;/&gt;
191 *   &lt;property name=&quot;ordered&quot; value=&quot;true&quot;/&gt;
192 *   &lt;property name=&quot;separated&quot; value=&quot;true&quot;/&gt;
193 *   &lt;property name=&quot;option&quot; value=&quot;above&quot;/&gt;
194 *   &lt;property name=&quot;sortStaticImportsAlphabetically&quot; value=&quot;true&quot;/&gt;
195 * &lt;/module&gt;
196 * </pre>
197 * <p>
198 * To configure the check so that it matches default IntelliJ IDEA formatter configuration
199 * (tested on v2018.2):
200 * </p>
201 * <ul>
202 * <li>
203 * group of static imports is on the bottom
204 * </li>
205 * <li>
206 * groups of type imports: all imports except of "javax" and "java", then "javax" and "java"
207 * </li>
208 * <li>
209 * imports will be sorted in the groups
210 * </li>
211 * <li>
212 * groups are separated by, at least, one blank line and aren't separated internally
213 * </li>
214 * </ul>
215 * <p>
216 * Note: a <a href="https://checkstyle.org/config_filters.html#SuppressionXpathSingleFilter">
217 * suppression xpath single filter</a> is needed because
218 * IDEA has no blank line between "javax" and "java".
219 * ImportOrder has a limitation by design to enforce an empty line between groups ("java", "javax").
220 * There is no flexibility to enforce empty lines between some groups and no empty lines between
221 * other groups.
222 * </p>
223 * <p>
224 * Note: "separated" option is disabled because IDEA default has blank line between "java" and
225 * static imports, and no blank line between "javax" and "java".
226 * </p>
227 * <pre>
228 * &lt;module name=&quot;ImportOrder&quot;&gt;
229 *   &lt;property name=&quot;groups&quot; value=&quot;*,javax,java&quot;/&gt;
230 *   &lt;property name=&quot;ordered&quot; value=&quot;true&quot;/&gt;
231 *   &lt;property name=&quot;separated&quot; value=&quot;false&quot;/&gt;
232 *   &lt;property name=&quot;option&quot; value=&quot;bottom&quot;/&gt;
233 *   &lt;property name=&quot;sortStaticImportsAlphabetically&quot; value=&quot;true&quot;/&gt;
234 * &lt;/module&gt;
235 * &lt;module name="SuppressionXpathSingleFilter"&gt;
236 *   &lt;property name="checks" value="ImportOrder"/&gt;
237 *   &lt;property name="message" value="^'java\..*'.*"/&gt;
238 * &lt;/module&gt;
239 * </pre>
240 * <p>
241 * To configure the check so that it matches default NetBeans formatter configuration
242 * (tested on v8):
243 * </p>
244 * <ul>
245 * <li>
246 * groups of type imports are not defined, all imports will be sorted as a one group
247 * </li>
248 * <li>
249 * static imports are not separated, they will be sorted along with other imports
250 * </li>
251 * </ul>
252 * <pre>
253 * &lt;module name=&quot;ImportOrder&quot;&gt;
254 *   &lt;property name=&quot;option&quot; value=&quot;inflow&quot;/&gt;
255 * &lt;/module&gt;
256 * </pre>
257 * <p>
258 * Group descriptions enclosed in slashes are interpreted as regular expressions.
259 * If multiple groups match, the one matching a longer substring of the imported name
260 * will take precedence, with ties broken first in favor of earlier matches and finally
261 * in favor of the first matching group.
262 * </p>
263 * <p>
264 * There is always a wildcard group to which everything not in a named group belongs.
265 * If an import does not match a named group, the group belongs to this wildcard group.
266 * The wildcard group position can be specified using the {@code *} character.
267 * </p>
268 * <p>
269 * Check also has on option making it more flexible: <b>sortStaticImportsAlphabetically</b>
270 * - sets whether static imports grouped by <b>top</b> or <b>bottom</b> option should be sorted
271 * alphabetically or not, default value is <b>false</b>. It is applied to static imports grouped
272 * with <b>top</b> or <b>bottom</b> options. This option is helping in reconciling of this
273 * Check and other tools like Eclipse's Organize Imports feature.
274 * </p>
275 * <p>
276 * To configure the Check allows static imports grouped to the <b>top</b> being sorted
277 * alphabetically:
278 * </p>
279 * <pre>
280 * &lt;module name=&quot;ImportOrder&quot;&gt;
281 *   &lt;property name=&quot;sortStaticImportsAlphabetically&quot; value=&quot;true&quot;/&gt;
282 *   &lt;property name=&quot;option&quot; value=&quot;top&quot;/&gt;
283 * &lt;/module&gt;
284 * </pre>
285 * <pre>
286 * import static java.lang.Math.PI;
287 * import static java.lang.Math.abs; // OK, alphabetical case sensitive ASCII order, 'P' &lt; 'a'
288 * import static org.abego.treelayout.Configuration.AlignmentInLevel; // OK, alphabetical order
289 *
290 * import org.abego.*;
291 *
292 * import java.util.Set; //  Wrong order for 'java.util.Set' import.
293 *
294 * public class SomeClass { ... }
295 * </pre>
296 * <p>
297 * To configure the Check with groups of static imports:
298 * </p>
299 * <pre>
300 * &lt;module name=&quot;ImportOrder&quot;&gt;
301 *   &lt;property name=&quot;staticGroups&quot; value=&quot;org,java&quot;/&gt;
302 *   &lt;property name=&quot;sortStaticImportsAlphabetically&quot; value=&quot;true&quot;/&gt;
303 * &lt;/module&gt;
304 * </pre>
305 * <pre>
306 * import static org.abego.treelayout.Configuration.AlignmentInLevel; // Group 1
307 * import static java.lang.Math.abs; // Group 2
308 * import static java.lang.String.format; // Group 2
309 * import static com.google.common.primitives.Doubles.BYTES; // Group "everything else"
310 *
311 * public class SomeClass { ... }
312 * </pre>
313 * <p>
314 * The following example shows the idea of 'useContainerOrderingForStatic' option that is
315 * useful for Eclipse IDE users to match ordering validation.
316 * This is how the import comparison works for static imports: we first compare
317 * the container of the static import, container is the type enclosing the static element
318 * being imported. When the result of the comparison is 0 (containers are equal),
319 * we compare the fully qualified import names.
320 * For e.g. this is what is considered to be container names for the given example:
321 *
322 * import static HttpConstants.COLON     =&gt; HttpConstants
323 * import static HttpHeaders.addHeader   =&gt; HttpHeaders
324 * import static HttpHeaders.setHeader   =&gt; HttpHeaders
325 * import static HttpHeaders.Names.DATE  =&gt; HttpHeaders.Names
326 *
327 * According to this logic, HttpHeaders.Names should come after HttpHeaders.
328 * </p>
329 * <p>
330 * Example for {@code useContainerOrderingForStatic=true}
331 * </p>
332 * <pre>
333 * &lt;module name=&quot;ImportOrder&quot;&gt;
334 *   &lt;property name=&quot;useContainerOrderingForStatic&quot; value=&quot;true&quot;/&gt;
335 *   &lt;property name=&quot;ordered&quot; value=&quot;true&quot;/&gt;
336 *   &lt;property name=&quot;option&quot; value=&quot;top&quot;/&gt;
337 *   &lt;property name=&quot;caseSensitive&quot; value=&quot;false&quot;/&gt;
338 *   &lt;property name=&quot;sortStaticImportsAlphabetically&quot; value=&quot;true&quot;/&gt;
339 * &lt;/module&gt;
340 * </pre>
341 * <pre>
342 * import static io.netty.handler.codec.http.HttpConstants.COLON;
343 * import static io.netty.handler.codec.http.HttpHeaders.addHeader;
344 * import static io.netty.handler.codec.http.HttpHeaders.setHeader;
345 * import static io.netty.handler.codec.http.HttpHeaders.Names.DATE;
346 *
347 * public class InputEclipseStaticImportsOrder { }
348 * </pre>
349 * <p>
350 * Example for {@code useContainerOrderingForStatic=false}
351 * </p>
352 * <pre>
353 * &lt;module name=&quot;ImportOrder&quot;&gt;
354 *   &lt;property name=&quot;useContainerOrderingForStatic&quot; value=&quot;false&quot;/&gt;
355 *   &lt;property name=&quot;ordered&quot; value=&quot;true&quot;/&gt;
356 *   &lt;property name=&quot;option&quot; value=&quot;top&quot;/&gt;
357 *   &lt;property name=&quot;caseSensitive&quot; value=&quot;false&quot;/&gt;
358 *   &lt;property name=&quot;sortStaticImportsAlphabetically&quot; value=&quot;true&quot;/&gt;
359 * &lt;/module&gt;
360 * </pre>
361 * <pre>
362 * import static io.netty.handler.codec.http.HttpConstants.COLON;
363 * import static io.netty.handler.codec.http.HttpHeaders.addHeader;
364 * import static io.netty.handler.codec.http.HttpHeaders.setHeader;
365 * import static io.netty.handler.codec.http.HttpHeaders.Names.DATE; // violation
366 *
367 * public class InputEclipseStaticImportsOrder { }
368 * </pre>
369 *
370 * @since 3.2
371 */

604    @Override
605    public void beginTree(DetailAST rootAST) {
606        lastGroup = Integer.MIN_VALUE;
607        lastImportLine = Integer.MIN_VALUE;
608        lastImport = "";
609        lastImportStatic = false;
610        beforeFirstImport = true;
611        staticImportsApart =
612            option == ImportOrderOption.TOP || option == ImportOrderOption.BOTTOM;
613    }

615    // -@cs[CyclomaticComplexity] SWITCH was transformed into IF-ELSE.
616    @Override
617    public void visitToken(DetailAST ast) {
618        final FullIdent ident;
619        final boolean isStatic;
620
621        if (ast.getType() == TokenTypes.IMPORT) {
622            ident = FullIdent.createFullIdentBelow(ast);
623            isStatic = false;
624        }
625        else {
626            ident = FullIdent.createFullIdent(ast.getFirstChild()
627                    .getNextSibling());
628            isStatic = true;
629        }
630
631        // using set of IF instead of SWITCH to analyze Enum options to satisfy coverage.
632        // https://github.com/checkstyle/checkstyle/issues/1387
633        if (option == ImportOrderOption.TOP || option == ImportOrderOption.ABOVE) {
634            final boolean isStaticAndNotLastImport = isStatic && !lastImportStatic;
635            doVisitToken(ident, isStatic, isStaticAndNotLastImport, ast);
636        }
637        else if (option == ImportOrderOption.BOTTOM || option == ImportOrderOption.UNDER) {
638            final boolean isLastImportAndNonStatic = lastImportStatic && !isStatic;
639            doVisitToken(ident, isStatic, isLastImportAndNonStatic, ast);
640        }
641        else if (option == ImportOrderOption.INFLOW) {
642            // "previous" argument is useless here
643            doVisitToken(ident, isStatic, true, ast);
644        }
645        else {
646            throw new IllegalStateException(
647                    "Unexpected option for static imports: " + option);
648        }
649
650        lastImportLine = ast.findFirstToken(TokenTypes.SEMI).getLineNo();
651        lastImportStatic = isStatic;
652        beforeFirstImport = false;
653    }

655    /**
656     * Shares processing...
657     *
658     * @param ident the import to process.
659     * @param isStatic whether the token is static or not.
660     * @param previous previous non-static but current is static (above), or
661     *                  previous static but current is non-static (under).
662     * @param ast node of the AST.
663     */
664    private void doVisitToken(FullIdent ident, boolean isStatic, boolean previous, DetailAST ast) {
665        final String name = ident.getText();
666        final int groupIdx = getGroupNumber(isStatic && staticImportsApart, name);
667
668        if (groupIdx > lastGroup) {
669            if (!beforeFirstImport
670                && ast.getLineNo() - lastImportLine < 2
671                && needSeparator(isStatic)) {
672                log(ast, MSG_SEPARATION, name);
673            }
674        }
675        else if (groupIdx == lastGroup) {
676            doVisitTokenInSameGroup(isStatic, previous, name, ast);
677        }
678        else {
679            log(ast, MSG_ORDERING, name);
680        }
681        if (isSeparatorInGroup(groupIdx, isStatic, ast.getLineNo())) {
682            log(ast, MSG_SEPARATED_IN_GROUP, name);
683        }
684
685        lastGroup = groupIdx;
686        lastImport = name;
687    }

689    /**
690     * Checks whether import groups should be separated.
691     *
692     * @param isStatic whether the token is static or not.
693     * @return true if imports groups should be separated.
694     */
695    private boolean needSeparator(boolean isStatic) {
696        final boolean typeImportSeparator = !isStatic && separated;
697        final boolean staticImportSeparator;
698        if (staticImportsApart) {
699            staticImportSeparator = isStatic && separatedStaticGroups;
700        }
701        else {
702            staticImportSeparator = isStatic && separated;
703        }
704        final boolean separatorBetween = isStatic != lastImportStatic
705            && (separated || separatedStaticGroups);
706
707        return typeImportSeparator || staticImportSeparator || separatorBetween;
708    }

710    /**
711     * Checks whether imports group separated internally.
712     *
713     * @param groupIdx group number.
714     * @param isStatic whether the token is static or not.
715     * @param line the line of the current import.
716     * @return true if imports group are separated internally.
717     */
718    private boolean isSeparatorInGroup(int groupIdx, boolean isStatic, int line) {
719        final boolean inSameGroup = groupIdx == lastGroup;
720        return (inSameGroup || !needSeparator(isStatic)) && isSeparatorBeforeImport(line);
721    }

723    /**
724     * Checks whether there is any separator before current import.
725     *
726     * @param line the line of the current import.
727     * @return true if there is separator before current import which isn't the first import.
728     */
729    private boolean isSeparatorBeforeImport(int line) {
730        return !beforeFirstImport && line - lastImportLine > 1;
731    }

733    /**
734     * Shares processing...
735     *
736     * @param isStatic whether the token is static or not.
737     * @param previous previous non-static but current is static (above), or
738     *     previous static but current is non-static (under).
739     * @param name the name of the current import.
740     * @param ast node of the AST.
741     */
742    private void doVisitTokenInSameGroup(boolean isStatic,
743            boolean previous, String name, DetailAST ast) {
744        if (ordered) {
745            if (option == ImportOrderOption.INFLOW) {
746                if (isWrongOrder(name, isStatic)) {
747                    log(ast, MSG_ORDERING, name);
748                }
749            }
750            else {
751                final boolean shouldFireError =
752                    // previous non-static but current is static (above)
753                    // or
754                    // previous static but current is non-static (under)
755                    previous
756                        ||
757                        // current and previous static or current and
758                        // previous non-static
759                        lastImportStatic == isStatic
760                    && isWrongOrder(name, isStatic);
761
762                if (shouldFireError) {
763                    log(ast, MSG_ORDERING, name);
764                }
765            }
766        }
767    }

769    /**
770     * Checks whether import name is in wrong order.
771     *
772     * @param name import name.
773     * @param isStatic whether it is a static import name.
774     * @return true if import name is in wrong order.
775     */
776    private boolean isWrongOrder(String name, boolean isStatic) {
777        final boolean result;
778        if (isStatic) {
779            if (useContainerOrderingForStatic) {
780                result = compareContainerOrder(lastImport, name, caseSensitive) > 0;
781            }
782            else if (staticImportsApart) {
783                result = sortStaticImportsAlphabetically
784                    && compare(lastImport, name, caseSensitive) > 0;
785            }
786            else {
787                result = compare(lastImport, name, caseSensitive) > 0;
788            }
789        }
790        else {
791            // out of lexicographic order
792            result = compare(lastImport, name, caseSensitive) > 0;
793        }
794        return result;
795    }

797    /**
798     * Compares two import strings.
799     * We first compare the container of the static import, container being the type enclosing
800     * the static element being imported. When this returns 0, we compare the qualified
801     * import name. For e.g. this is what is considered to be container names:
802     * <p>
803     * import static HttpConstants.COLON     => HttpConstants
804     * import static HttpHeaders.addHeader   => HttpHeaders
805     * import static HttpHeaders.setHeader   => HttpHeaders
806     * import static HttpHeaders.Names.DATE  => HttpHeaders.Names
807     * </p>
808     * <p>
809     * According to this logic, HttpHeaders.Names would come after HttpHeaders.
810     *
811     * For more details, see <a href="https://bugs.eclipse.org/bugs/show_bug.cgi?id=473629#c3">
812     * static imports comparison method</a> in Eclipse.
813     * </p>
814     *
815     * @param importName1 first import name.
816     * @param importName2 second import name.
817     * @param caseSensitive whether the comparison of fully qualified import names is case
818     *                      sensitive.
819     * @return the value {@code 0} if str1 is equal to str2; a value
820     *         less than {@code 0} if str is less than the str2 (container order
821     *         or lexicographical); and a value greater than {@code 0} if str1 is greater than str2
822     *         (container order or lexicographically).
823     */
824    private static int compareContainerOrder(String importName1, String importName2,
825                                             boolean caseSensitive) {
826        final String container1 = getImportContainer(importName1);
827        final String container2 = getImportContainer(importName2);
828        final int compareContainersOrderResult;
829        if (caseSensitive) {
830            compareContainersOrderResult = container1.compareTo(container2);
831        }
832        else {
833            compareContainersOrderResult = container1.compareToIgnoreCase(container2);
834        }
835        final int result;
836        if (compareContainersOrderResult == 0) {
837            result = compare(importName1, importName2, caseSensitive);
838        }
839        else {
840            result = compareContainersOrderResult;
841        }
842        return result;
843    }

845    /**
846     * Extracts import container name from fully qualified import name.
847     * An import container name is the type which encloses the static element being imported.
848     * For example, HttpConstants, HttpHeaders, HttpHeaders.Names are import container names:
849     * <p>
850     * import static HttpConstants.COLON     => HttpConstants
851     * import static HttpHeaders.addHeader   => HttpHeaders
852     * import static HttpHeaders.setHeader   => HttpHeaders
853     * import static HttpHeaders.Names.DATE  => HttpHeaders.Names
854     * </p>
855     *
856     * @param qualifiedImportName fully qualified import name.
857     * @return import container name.
858     */
859    private static String getImportContainer(String qualifiedImportName) {
860        final int lastDotIndex = qualifiedImportName.lastIndexOf('.');
861        return qualifiedImportName.substring(0, lastDotIndex);
862    }

864    /**
865     * Finds out what group the specified import belongs to.
866     *
867     * @param isStatic whether the token is static or not.
868     * @param name the import name to find.
869     * @return group number for given import name.
870     */
871    private int getGroupNumber(boolean isStatic, String name) {
872        final Pattern[] patterns;
873        if (isStatic) {
874            patterns = staticGroups;
875        }
876        else {
877            patterns = groups;
878        }
879
880        int number = getGroupNumber(patterns, name);
881
882        if (isStatic && option == ImportOrderOption.BOTTOM) {
883            number += groups.length + 1;
884        }
885        else if (!isStatic && option == ImportOrderOption.TOP) {
886            number += staticGroups.length + 1;
887        }
888        return number;
889    }

891    /**
892     * Finds out what group the specified import belongs to.
893     *
894     * @param patterns groups to check.
895     * @param name the import name to find.
896     * @return group number for given import name.
897     */
898    private static int getGroupNumber(Pattern[] patterns, String name) {
899        int bestIndex = patterns.length;
900        int bestEnd = -1;
901        int bestPos = Integer.MAX_VALUE;
902
903        // find out what group this belongs in
904        // loop over patterns and get index
905        for (int i = 0; i < patterns.length; i++) {
906            final Matcher matcher = patterns[i].matcher(name);
907            if (matcher.find()) {
908                if (matcher.start() < bestPos) {
909                    bestIndex = i;
910                    bestEnd = matcher.end();
911                    bestPos = matcher.start();
912                }
913                else if (matcher.start() == bestPos && matcher.end() > bestEnd) {
914                    bestIndex = i;
915                    bestEnd = matcher.end();
916                }
917            }
918        }
919        return bestIndex;
920    }

922    /**
923     * Compares two strings.
924     *
925     * @param string1
926     *            the first string.
927     * @param string2
928     *            the second string.
929     * @param caseSensitive
930     *            whether the comparison is case sensitive.
931     * @return the value {@code 0} if string1 is equal to string2; a value
932     *         less than {@code 0} if string1 is lexicographically less
933     *         than the string2; and a value greater than {@code 0} if
934     *         string1 is lexicographically greater than string2.
935     */
936    private static int compare(String string1, String string2,
937            boolean caseSensitive) {
938        final int result;
939        if (caseSensitive) {
940            result = string1.compareTo(string2);
941        }
942        else {
943            result = string1.compareToIgnoreCase(string2);
944        }
945
946        return result;
947    }

949    /**
950     * Compiles the list of package groups and the order they should occur in the file.
951     *
952     * @param packageGroups a comma-separated list of package names/prefixes.
953     * @return array of compiled patterns.
954     */
955    private static Pattern[] compilePatterns(String... packageGroups) {
956        final Pattern[] patterns = new Pattern[packageGroups.length];
957
958        for (int i = 0; i < packageGroups.length; i++) {
959            String pkg = packageGroups[i];
960            final Pattern grp;
961
962            // if the pkg name is the wildcard, make it match zero chars
963            // from any name, so it will always be used as last resort.
964            if (WILDCARD_GROUP_NAME.equals(pkg)) {
965                // matches any package
966                grp = Pattern.compile("");
967            }
968            else if (CommonUtil.startsWithChar(pkg, '/')) {
969                if (!CommonUtil.endsWithChar(pkg, '/')) {
970                    throw new IllegalArgumentException("Invalid group: " + pkg);
971                }
972                pkg = pkg.substring(1, pkg.length() - 1);
973                grp = Pattern.compile(pkg);
974            }
975            else {
976                final StringBuilder pkgBuilder = new StringBuilder(pkg);
977                if (!CommonUtil.endsWithChar(pkg, '.')) {
978                    pkgBuilder.append('.');
979                }
980                grp = Pattern.compile("^" + Pattern.quote(pkgBuilder.toString()));
981            }
982
983            patterns[i] = grp;
984        }
985        return patterns;
986    }
