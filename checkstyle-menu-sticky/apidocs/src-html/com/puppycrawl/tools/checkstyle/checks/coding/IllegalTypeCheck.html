037/**
038 * <p>
039 * Checks that particular classes or interfaces are never used.
040 * </p>
041 * <p>
042 * Rationale: Helps reduce coupling on concrete classes.
043 * </p>
044 * <p>
045 * For additional restriction of type usage see also:
046 * <a href="https://checkstyle.org/config_coding.html#IllegalInstantiation">
047 * IllegalInstantiation</a>,
048 * <a href="https://checkstyle.org/config_imports.html#IllegalImport">IllegalImport</a>
049 * </p>
050 * <p>
051 * It is possible to set illegal class names via short or
052 * <a href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-6.html#jls-6.7">canonical</a>
053 * name. Specifying illegal type invokes analyzing imports and Check puts violations at
054 * corresponding declarations (of variables, methods or parameters).
055 * This helps to avoid ambiguous cases, e.g.: {@code java.awt.List} was set as
056 * illegal class name, then, code like:
057 * </p>
058 * <pre>
059 * import java.util.List;
060 * ...
061 * List list; //No violation here
062 * </pre>
063 * <p>
064 * will be ok.
065 * </p>
066 * <p>
067 * In most cases it's justified to put following classes to <b>illegalClassNames</b>:
068 * </p>
069 * <ul>
070 * <li>GregorianCalendar</li>
071 * <li>Hashtable</li>
072 * <li>ArrayList</li>
073 * <li>LinkedList</li>
074 * <li>Vector</li>
075 * </ul>
076 * <p>
077 * as methods that are differ from interface methods are rarely used, so in most cases user will
078 * benefit from checking for them.
079 * </p>
080 * <ul>
081 * <li>
082 * Property {@code validateAbstractClassNames} - Control whether to validate abstract class names.
083 * Default value is {@code false}.
084 * </li>
085 * <li>
086 * Property {@code illegalClassNames} - Specify classes that should not be used
087 * as types in variable declarations, return values or parameters.
088 * Default value is {@code HashMap, HashSet, LinkedHashMap, LinkedHashSet, TreeMap,
089 * TreeSet, java.util.HashMap, java.util.HashSet, java.util.LinkedHashMap,
090 * java.util.LinkedHashSet, java.util.TreeMap, java.util.TreeSet}.
091 * </li>
092 * <li>
093 * Property {@code legalAbstractClassNames} - Define abstract classes that may be used as types.
094 * Default value is {@code {}}.
095 * </li>
096 * <li>
097 * Property {@code ignoredMethodNames} - Specify methods that should not be checked.
098 * Default value is {@code getEnvironment, getInitialContext}.
099 * </li>
100 * <li>
101 * Property {@code illegalAbstractClassNameFormat} - Specify RegExp for illegal abstract class
102 * names.
103 * Default value is {@code "^(.*[.])?Abstract.*$"}.
104 * </li>
105 * <li>
106 * Property {@code memberModifiers} - Control whether to check only methods and fields with any
107 * of the specified modifiers.
108 * This property does not affect method calls nor method references.
109 * Default value is no tokens.
110 * </li>
111 * <li>
112 * Property {@code tokens} - tokens to check
113 * Default value is:
114 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_FIELD_DEF">
115 * ANNOTATION_FIELD_DEF</a>,
116 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
117 * CLASS_DEF</a>,
118 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
119 * INTERFACE_DEF</a>,
120 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_CALL">
121 * METHOD_CALL</a>,
122 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
123 * METHOD_DEF</a>,
124 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_REF">
125 * METHOD_REF</a>,
126 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#PARAMETER_DEF">
127 * PARAMETER_DEF</a>,
128 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#VARIABLE_DEF">
129 * VARIABLE_DEF</a>.
130 * </li>
131 * </ul>
132 * <p>
133 * Default Configuration:
134 * </p>
135 * <pre>
136 * &lt;module name=&quot;IllegalType&quot;/&gt;
137 * </pre>
138 * <pre>
139 * public class Test extends TreeSet { // violation
140 *   public &lt;T extends java.util.HashSet&gt; void method() { // violation
141 *
142 *     LinkedHashMap&lt;Integer, String&gt; lhmap =
143 *     new LinkedHashMap&lt;Integer, String&gt;(); // violation
144 *     TreeMap&lt;Integer, String&gt; treemap =
145 *     new TreeMap&lt;Integer, String&gt;(); // violation
146 *     Test t; // OK
147 *     HashMap&lt;String, String&gt; hmap; // violation
148 *     Queue&lt;Integer&gt; intqueue; // OK
149 *
150 *     java.lang.IllegalArgumentException illegalex; // OK
151 *     java.util.TreeSet treeset; // violation
152 *   }
153 *
154 * }
155 * </pre>
156 * <p>
157 * To configure the Check so that particular tokens are checked:
158 * </p>
159 * <pre>
160 * &lt;module name="IllegalType"&gt;
161 *   &lt;property name="tokens" value="METHOD_DEF"/&gt;
162 * &lt;/module&gt;
163 * </pre>
164 * <pre>
165 * public class Test extends TreeSet { // OK
166 *   public &lt;T extends java.util.HashSet&gt; void method() { // violation
167 *     LinkedHashMap&lt;Integer, String&gt; lhmap =
168 *     new LinkedHashMap&lt;Integer, String&gt;(); // OK
169 *
170 *     java.lang.IllegalArgumentException illegalex; // OK
171 *     java.util.TreeSet treeset; // Ok
172 *   }
173 *
174 *   public &lt;T extends java.util.HashSet&gt; void typeParam(T t) {} // violation
175 *
176 *   public void fullName(TreeSet a) {} // OK
177 *
178 * }
179 * </pre>
180 * <p>
181 * To configure the Check so that it ignores function() methods:
182 * </p>
183 * <pre>
184 * &lt;module name=&quot;IllegalType&quot;&gt;
185 *   &lt;property name=&quot;ignoredMethodNames&quot; value=&quot;function&quot;/&gt;
186 * &lt;/module&gt;
187 * </pre>
188 * <pre>
189 * public class Test {
190 *   public HashMap&lt;String, String&gt; function() { // OK
191 *     // code
192 *   }
193 *
194 *   public HashMap&lt;String, String&gt; function1() { // violation
195 *     // code
196 *   }
197 * }
198 * </pre>
199 * <p>
200 * To configure the Check so that it validates abstract class names:
201 * </p>
202 * <pre>
203 *  &lt;module name=&quot;IllegalType&quot;&gt;
204 *    &lt;property name=&quot;validateAbstractClassNames&quot; value=&quot;true&quot;/&gt;
205 *    &lt;property name=&quot;illegalAbstractClassNameFormat&quot; value=&quot;Gitt&quot;/&gt;
206 *  &lt;/module&gt;
207 * </pre>
208 * <pre>
209 * class Test extends Gitter { // violation
210 * }
211 *
212 * class Test1 extends Github { // OK
213 * }
214 * </pre>
215 * <p>
216 * To configure the Check so that it verifies only public, protected or static methods and fields:
217 * </p>
218 * <pre>
219 * &lt;module name=&quot;IllegalType&quot;&gt;
220 *   &lt;property name=&quot;memberModifiers&quot; value=&quot;LITERAL_PUBLIC,
221 *    LITERAL_PROTECTED, LITERAL_STATIC&quot;/&gt;
222 * &lt;/module&gt;
223 * </pre>
224 * <pre>
225 * public class Test {
226 *   public HashMap&lt;String, String&gt; function1() { // violation
227 *     // code
228 *   }
229 *
230 *   private HashMap&lt;String, String&gt; function2() { // OK
231 *     // code
232 *   }
233 *
234 *   protected HashMap&lt;Integer, String&gt; function3() { // violation
235 *     // code
236 *   }
237 *
238 *   public static TreeMap&lt;Integer, String&gt; function4() { // violation
239 *     // code
240 *   }
241 *
242 * }
243 * </pre>
244 * <p>
245 * To configure the check so that it verifies usage of types Boolean and Foo:
246 * </p>
247 * <pre>
248 * &lt;module name=&quot;IllegalType&quot;&gt;
249 *           &lt;property name=&quot;illegalClassNames&quot; value=&quot;Boolean, Foo&quot;/&gt;
250 * &lt;/module&gt;
251 * </pre>
252 * <pre>
253 * public class Test {
254 *
255 *   public Set&lt;Boolean&gt; set; // violation
256 *   public java.util.List&lt;Map&lt;Boolean, Foo&gt;&gt; list; // violation
257 *
258 *   private void method(List&lt;Foo&gt; list, Boolean value) { // violation
259 *     SomeType.&lt;Boolean&gt;foo(); // violation
260 *     final Consumer&lt;Foo&gt; consumer = Foo&lt;Boolean&gt;::foo; // violation
261 *   }
262 *
263 *   public &lt;T extends Boolean, U extends Serializable&gt; void typeParam(T a) {} // violation
264 *
265 *   public void fullName(java.util.ArrayList&lt;? super Boolean&gt; a) {} // violation
266 *
267 *   public abstract Set&lt;Boolean&gt; shortName(Set&lt;? super Boolean&gt; a); // violation
268 *
269 *   public Set&lt;? extends Foo&gt; typeArgument() { // violation
270 *     return new TreeSet&lt;Foo&lt;Boolean&gt;&gt;();
271 *   }
272 *
273 * }
274 * </pre>
275 *
276 * @since 3.2
277 *
278 */
279@FileStatefulCheck
280public final class IllegalTypeCheck extends AbstractCheck {

282    /**
283     * A key is pointing to the warning message text in "messages.properties"
284     * file.
285     */
286    public static final String MSG_KEY = "illegal.type";
287
288    /** Types illegal by default. */
289    private static final String[] DEFAULT_ILLEGAL_TYPES = {
290        "HashSet",
291        "HashMap",
292        "LinkedHashMap",
293        "LinkedHashSet",
294        "TreeSet",
295        "TreeMap",
296        "java.util.HashSet",
297        "java.util.HashMap",
298        "java.util.LinkedHashMap",
299        "java.util.LinkedHashSet",
300        "java.util.TreeSet",
301        "java.util.TreeMap",
302    };
303
304    /** Default ignored method names. */
305    private static final String[] DEFAULT_IGNORED_METHOD_NAMES = {
306        "getInitialContext",
307        "getEnvironment",
308    };
309
310    /**
311     * Specify classes that should not be used as types in variable declarations,
312     * return values or parameters.
313     */
314    private final Set<String> illegalClassNames = new HashSet<>();
315    /** Illegal short classes. */
316    private final Set<String> illegalShortClassNames = new HashSet<>();
317    /** Define abstract classes that may be used as types. */
318    private final Set<String> legalAbstractClassNames = new HashSet<>();
319    /** Specify methods that should not be checked. */
320    private final Set<String> ignoredMethodNames = new HashSet<>();
321    /**
322     * Control whether to check only methods and fields with any of the specified modifiers.
323     * This property does not affect method calls nor method references.
324     */
325    private List<Integer> memberModifiers;
326
327    /** Specify RegExp for illegal abstract class names. */
328    private Pattern illegalAbstractClassNameFormat = Pattern.compile("^(.*[.])?Abstract.*$");
329
330    /**
331     * Control whether to validate abstract class names.
332     */
333    private boolean validateAbstractClassNames;
334
335    /** Creates new instance of the check. */
336    public IllegalTypeCheck() {
337        setIllegalClassNames(DEFAULT_ILLEGAL_TYPES);
338        setIgnoredMethodNames(DEFAULT_IGNORED_METHOD_NAMES);
339    }

341    /**
342     * Setter to specify RegExp for illegal abstract class names.
343     *
344     * @param pattern a pattern.
345     */
346    public void setIllegalAbstractClassNameFormat(Pattern pattern) {
347        illegalAbstractClassNameFormat = pattern;
348    }

350    /**
351     * Setter to control whether to validate abstract class names.
352     *
353     * @param validateAbstractClassNames whether abstract class names must be ignored.

355    public void setValidateAbstractClassNames(boolean validateAbstractClassNames) {
356        this.validateAbstractClassNames = validateAbstractClassNames;

359    @Override
360    public int[] getDefaultTokens() {
361        return getAcceptableTokens();
362    }
363
364    @Override
365    public int[] getAcceptableTokens() {
366        return new int[] {
367            TokenTypes.ANNOTATION_FIELD_DEF,
368            TokenTypes.CLASS_DEF,
369            TokenTypes.IMPORT,
370            TokenTypes.INTERFACE_DEF,
371            TokenTypes.METHOD_CALL,
372            TokenTypes.METHOD_DEF,
373            TokenTypes.METHOD_REF,
374            TokenTypes.PARAMETER_DEF,
375            TokenTypes.VARIABLE_DEF,
376        };
377    }

379    @Override
380    public void beginTree(DetailAST rootAST) {
381        illegalShortClassNames.clear();
382
383        for (String s : illegalClassNames) {
384            if (s.indexOf('.') == -1) {
385                illegalShortClassNames.add(s);
386            }
387        }
388    }
389
390    @Override
391    public int[] getRequiredTokens() {
392        return new int[] {TokenTypes.IMPORT};
393    }
394
395    @Override
396    public void visitToken(DetailAST ast) {
397        switch (ast.getType()) {
398            case TokenTypes.CLASS_DEF:
399            case TokenTypes.INTERFACE_DEF:
400                visitTypeDef(ast);
401                break;
402            case TokenTypes.METHOD_CALL:
403            case TokenTypes.METHOD_REF:
404                visitMethodCallOrRef(ast);
405                break;
406            case TokenTypes.METHOD_DEF:
407                visitMethodDef(ast);
408                break;
409            case TokenTypes.VARIABLE_DEF:
410            case TokenTypes.ANNOTATION_FIELD_DEF:
411                visitVariableDef(ast);
412                break;
413            case TokenTypes.PARAMETER_DEF:
414                visitParameterDef(ast);
415                break;
416            case TokenTypes.IMPORT:
417                visitImport(ast);
418                break;
419            default:
420                throw new IllegalStateException(ast.toString());
421        }
422    }
423
424    /**
425     * Checks if current method's return type or variable's type is verifiable
426     * according to <b>memberModifiers</b> option.
427     *
428     * @param methodOrVariableDef METHOD_DEF or VARIABLE_DEF ast node.
429     * @return true if member is verifiable according to <b>memberModifiers</b> option.
430     */
431    private boolean isVerifiable(DetailAST methodOrVariableDef) {
432        boolean result = true;
433        if (memberModifiers != null) {
434            final DetailAST modifiersAst = methodOrVariableDef
435                    .findFirstToken(TokenTypes.MODIFIERS);
436            result = isContainVerifiableType(modifiersAst);
437        }
438        return result;
439    }
440
441    /**
442     * Checks is modifiers contain verifiable type.
443     *
444     * @param modifiers
445     *            parent node for all modifiers
446     * @return true if method or variable can be verified
447     */
448    private boolean isContainVerifiableType(DetailAST modifiers) {
449        boolean result = false;
450        if (modifiers.getFirstChild() != null) {
451            for (DetailAST modifier = modifiers.getFirstChild(); modifier != null;
452                     modifier = modifier.getNextSibling()) {
453                if (memberModifiers.contains(modifier.getType())) {
454                    result = true;
455                    break;
456                }
457            }
458        }
459        return result;
460    }
461
462    /**
463     * Checks the super type and implemented interfaces of a given type.
464     *
465     * @param typeDef class or interface for check.
466     */
467    private void visitTypeDef(DetailAST typeDef) {
468        if (isVerifiable(typeDef)) {
469            checkTypeParameters(typeDef);
470            final DetailAST extendsClause = typeDef.findFirstToken(TokenTypes.EXTENDS_CLAUSE);
471            if (extendsClause != null) {
472                checkBaseTypes(extendsClause);
473            }
474            final DetailAST implementsClause = typeDef.findFirstToken(TokenTypes.IMPLEMENTS_CLAUSE);
475            if (implementsClause != null) {
476                checkBaseTypes(implementsClause);
477            }
478        }
479    }
480
481    /**
482     * Checks return type of a given method.
483     *
484     * @param methodDef method for check.
485     */
486    private void visitMethodDef(DetailAST methodDef) {
487        if (isCheckedMethod(methodDef)) {
488            checkClassName(methodDef);
489        }
490    }
491
492    /**
493     * Checks type of parameters.
494     *
495     * @param parameterDef parameter list for check.
496     */
497    private void visitParameterDef(DetailAST parameterDef) {
498        final DetailAST grandParentAST = parameterDef.getParent().getParent();
499
500        if (grandParentAST.getType() == TokenTypes.METHOD_DEF && isCheckedMethod(grandParentAST)) {
501            checkClassName(parameterDef);
502        }
503    }
504
505    /**
506     * Checks type of given variable.
507     *
508     * @param variableDef variable to check.
509     */
510    private void visitVariableDef(DetailAST variableDef) {
511        if (isVerifiable(variableDef)) {
512            checkClassName(variableDef);
513        }
514    }
515
516    /**
517     * Checks the type arguments of given method call/reference.
518     *
519     * @param methodCallOrRef method call/reference to check.
520     */
521    private void visitMethodCallOrRef(DetailAST methodCallOrRef) {
522        checkTypeArguments(methodCallOrRef);
523    }
524
525    /**
526     * Checks imported type (as static and star imports are not supported by Check,
527     *  only type is in the consideration).<br>
528     * If this type is illegal due to Check's options - puts violation on it.
529     *
530     * @param importAst {@link TokenTypes#IMPORT Import}
531     */
532    private void visitImport(DetailAST importAst) {
533        if (!isStarImport(importAst)) {
534            final String canonicalName = getImportedTypeCanonicalName(importAst);
535            extendIllegalClassNamesWithShortName(canonicalName);
536        }
537    }
538
539    /**
540     * Checks if current import is star import. E.g.:
541     * <p>
542     * {@code
543     * import java.util.*;
544     * }
545     * </p>
546     *
547     * @param importAst {@link TokenTypes#IMPORT Import}
548     * @return true if it is star import
549     */
550    private static boolean isStarImport(DetailAST importAst) {
551        boolean result = false;
552        DetailAST toVisit = importAst;
553        while (toVisit != null) {
554            toVisit = getNextSubTreeNode(toVisit, importAst);
555            if (toVisit != null && toVisit.getType() == TokenTypes.STAR) {
556                result = true;
557                break;
558            }
559        }
560        return result;
561    }
562
563    /**
564     * Checks type and type arguments/parameters of given method, parameter, variable or
565     * method call/reference.
566     *
567     * @param ast node to check.
568     */
569    private void checkClassName(DetailAST ast) {
570        final DetailAST type = ast.findFirstToken(TokenTypes.TYPE);
571        checkType(type);
572        checkTypeParameters(ast);
573    }
574
575    /**
576     * Checks the identifier of the given type.
577     *
578     * @param type node to check.
579     */
580    private void checkIdent(DetailAST type) {
581        final FullIdent ident = FullIdent.createFullIdent(type);
582        if (isMatchingClassName(ident.getText())) {
583            log(ident.getDetailAst(), MSG_KEY, ident.getText());
584        }
585    }
586
587    /**
588     * Checks the {@code extends} or {@code implements} statement.
589     *
590     * @param clause DetailAST for either {@link TokenTypes#EXTENDS_CLAUSE} or
591     *               {@link TokenTypes#IMPLEMENTS_CLAUSE}
592     */
593    private void checkBaseTypes(DetailAST clause) {
594        DetailAST child = clause.getFirstChild();
595        while (child != null) {
596            if (child.getType() == TokenTypes.IDENT) {
597                checkIdent(child);
598            }
599            else if (child.getType() == TokenTypes.TYPE_ARGUMENTS) {
600                TokenUtil.forEachChild(child, TokenTypes.TYPE_ARGUMENT, this::checkType);
601            }
602            child = child.getNextSibling();
603        }
604    }
605
606    /**
607     * Checks the given type, its arguments and parameters.
608     *
609     * @param type node to check.
610     */
611    private void checkType(DetailAST type) {
612        checkIdent(type.getFirstChild());
613        checkTypeArguments(type);
614        checkTypeBounds(type);
615    }
616
617    /**
618     * Checks the upper and lower bounds for the given type.
619     *
620     * @param type node to check.
621     */
622    private void checkTypeBounds(DetailAST type) {
623        final DetailAST upperBounds = type.findFirstToken(TokenTypes.TYPE_UPPER_BOUNDS);
624        if (upperBounds != null) {
625            checkType(upperBounds);
626        }
627        final DetailAST lowerBounds = type.findFirstToken(TokenTypes.TYPE_LOWER_BOUNDS);
628        if (lowerBounds != null) {
629            checkType(lowerBounds);
630        }
631    }
632
633    /**
634     * Checks the type parameters of the node.
635     *
636     * @param node node to check.
637     */
638    private void checkTypeParameters(final DetailAST node) {
639        final DetailAST typeParameters = node.findFirstToken(TokenTypes.TYPE_PARAMETERS);
640        if (typeParameters != null) {
641            TokenUtil.forEachChild(typeParameters, TokenTypes.TYPE_PARAMETER, this::checkType);
642        }
643    }
644
645    /**
646     * Checks the type arguments of the node.
647     *
648     * @param node node to check.
649     */
650    private void checkTypeArguments(final DetailAST node) {
651        DetailAST typeArguments = node.findFirstToken(TokenTypes.TYPE_ARGUMENTS);
652        if (typeArguments == null) {
653            typeArguments = node.getFirstChild().findFirstToken(TokenTypes.TYPE_ARGUMENTS);
654        }
655
656        if (typeArguments != null) {
657            TokenUtil.forEachChild(typeArguments, TokenTypes.TYPE_ARGUMENT, this::checkType);
658        }
659    }
660
661    /**
662     * Returns true if given class name is one of illegal classes or else false.
663     *
664     * @param className class name to check.
665     * @return true if given class name is one of illegal classes
666     *         or if it matches to abstract class names pattern.
667     */
668    private boolean isMatchingClassName(String className) {
669        final String shortName = className.substring(className.lastIndexOf('.') + 1);
670        return illegalClassNames.contains(className)
671                || illegalShortClassNames.contains(shortName)
672                || validateAbstractClassNames
673                    && !legalAbstractClassNames.contains(className)
674                    && illegalAbstractClassNameFormat.matcher(className).find();
675    }
676
677    /**
678     * Extends illegal class names set via imported short type name.
679     *
680     * @param canonicalName
681     *  <a href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-6.html#jls-6.7">
682     *  Canonical</a> name of imported type.
683     */
684    private void extendIllegalClassNamesWithShortName(String canonicalName) {
685        if (illegalClassNames.contains(canonicalName)) {
686            final String shortName = canonicalName
687                .substring(canonicalName.lastIndexOf('.') + 1);
688            illegalShortClassNames.add(shortName);
689        }
690    }
691
692    /**
693     * Gets imported type's
694     * <a href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-6.html#jls-6.7">
695     *  canonical name</a>.
696     *
697     * @param importAst {@link TokenTypes#IMPORT Import}
698     * @return Imported canonical type's name.
699     */
700    private static String getImportedTypeCanonicalName(DetailAST importAst) {
701        final StringBuilder canonicalNameBuilder = new StringBuilder(256);
702        DetailAST toVisit = importAst;
703        while (toVisit != null) {
704            toVisit = getNextSubTreeNode(toVisit, importAst);
705            if (toVisit != null && toVisit.getType() == TokenTypes.IDENT) {
706                if (canonicalNameBuilder.length() > 0) {
707                    canonicalNameBuilder.append('.');
708                }
709                canonicalNameBuilder.append(toVisit.getText());
710            }
711        }
712        return canonicalNameBuilder.toString();
713    }

715    /**
716     * Gets the next node of a syntactical tree (child of a current node or
717     * sibling of a current node, or sibling of a parent of a current node).
718     *
719     * @param currentNodeAst Current node in considering
720     * @param subTreeRootAst SubTree root
721     * @return Current node after bypassing, if current node reached the root of a subtree
722     *        method returns null
723     */
724    private static DetailAST
725        getNextSubTreeNode(DetailAST currentNodeAst, DetailAST subTreeRootAst) {
726        DetailAST currentNode = currentNodeAst;
727        DetailAST toVisitAst = currentNode.getFirstChild();
728        while (toVisitAst == null) {
729            toVisitAst = currentNode.getNextSibling();
730            if (toVisitAst == null) {
731                if (currentNode.getParent().equals(subTreeRootAst)) {
732                    break;
733                }
734                currentNode = currentNode.getParent();
735            }
736        }
737        return toVisitAst;
738    }

740    /**
741     * Returns true if method has to be checked or false.
742     *
743     * @param ast method def to check.
744     * @return true if we should check this method.
745     */
746    private boolean isCheckedMethod(DetailAST ast) {
747        final String methodName =
748            ast.findFirstToken(TokenTypes.IDENT).getText();
749        return isVerifiable(ast) && !ignoredMethodNames.contains(methodName)
750                && !AnnotationUtil.containsAnnotation(ast, "Override");
751    }
