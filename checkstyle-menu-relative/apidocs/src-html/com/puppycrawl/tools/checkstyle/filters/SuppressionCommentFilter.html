039/**
040 * <p>
041 * Filter {@code SuppressionCommentFilter} uses pairs of comments to suppress audit events.
042 * </p>
043 * <p>
044 * Rationale:
045 * Sometimes there are legitimate reasons for violating a check. When
046 * this is a matter of the code in question and not personal
047 * preference, the best place to override the policy is in the code
048 * itself. Semi-structured comments can be associated with the check.
049 * This is sometimes superior to a separate suppressions file, which
050 * must be kept up-to-date as the source file is edited.
051 * </p>
052 * <p>
053 * Note that the suppression comment should be put before the violation.
054 * You can use more than one suppression comment each on separate line.
055 * </p>
056 * <p>
057 * Attention: This filter may only be specified within the TreeWalker module
058 * ({@code &lt;module name="TreeWalker"/&gt;}) and only applies to checks which are also
059 * defined within this module. To filter non-TreeWalker checks like {@code RegexpSingleline}, a
060 * <a href="https://checkstyle.org/config_filters.html#SuppressWithPlainTextCommentFilter">
061 * SuppressWithPlainTextCommentFilter</a> or similar filter must be used.
062 * </p>
063 * <p>
064 * {@code offCommentFormat} and {@code onCommentFormat} must have equal
065 * <a href="https://docs.oracle.com/en/java/javase/11/docs/api/java.base/java/util/regex/Matcher.html#groupCount()">
066 * paren counts</a>.
067 * </p>
068 * <p>
069 * SuppressionCommentFilter can suppress Checks that have Treewalker as parent module.
070 * </p>
071 * <ul>
072 * <li>
073 * Property {@code offCommentFormat} - Specify comment pattern to
074 * trigger filter to begin suppression.
075 * Default value is {@code "CHECKSTYLE:OFF"}.
076 * </li>
077 * <li>
078 * Property {@code onCommentFormat} - Specify comment pattern to trigger filter to end suppression.
079 * Default value is {@code "CHECKSTYLE:ON"}.
080 * </li>
081 * <li>
082 * Property {@code checkFormat} - Specify check pattern to suppress.
083 * Default value is {@code ".*"}.
084 * </li>
085 * <li>
086 * Property {@code messageFormat} - Specify message pattern to suppress.
087 * Default value is {@code null}.
088 * </li>
089 * <li>
090 * Property {@code idFormat} - Specify check ID pattern to suppress.
091 * Default value is {@code null}.
092 * </li>
093 * <li>
094 * Property {@code checkCPP} - Control whether to check C++ style comments ({@code //}).
095 * Default value is {@code true}.
096 * </li>
097 * <li>
098 * Property {@code checkC} - Control whether to check C style comments ({@code &#47;* ... *&#47;}).
099 * Default value is {@code true}.
100 * </li>
101 * </ul>
102 * <p>
103 * To configure a filter to suppress audit events between a comment containing
104 * {@code CHECKSTYLE:OFF} and a comment containing {@code CHECKSTYLE:ON}:
105 * </p>
106 * <pre>
107 * &lt;module name="TreeWalker"&gt;
108 *               ...
109 *   &lt;module name="SuppressionCommentFilter"/&gt;
110 *               ...
111 * &lt;/module&gt;
112 * </pre>
113 * <p>
114 * To configure a filter to suppress audit events between a comment containing line
115 * {@code BEGIN GENERATED CODE} and a comment containing line {@code END GENERATED CODE}:
116 * </p>
117 * <pre>
118 * &lt;module name="SuppressionCommentFilter"&gt;
119 *   &lt;property name="offCommentFormat" value="BEGIN GENERATED CODE"/&gt;
120 *   &lt;property name="onCommentFormat" value="END GENERATED CODE"/&gt;
121 * &lt;/module&gt;
122 * </pre>
123 * <pre>
124 * //BEGIN GENERATED CODE
125 * &#64;Override
126 * public boolean equals(Object obj) { ... } // No violation events will be reported
127 *
128 * &#64;Override
129 * public int hashCode() { ... } // No violation events will be reported
130 * //END GENERATED CODE
131 * . . .
132 * </pre>
133 * <p>
134 * To configure a filter so that {@code // stop constant check} and
135 * {@code // resume constant check} marks legitimate constant names:
136 * </p>
137 * <pre>
138 * &lt;module name="SuppressionCommentFilter"&gt;
139 *   &lt;property name="offCommentFormat" value="stop constant check"/&gt;
140 *   &lt;property name="onCommentFormat" value="resume constant check"/&gt;
141 *   &lt;property name="checkFormat" value="ConstantNameCheck"/&gt;
142 * &lt;/module&gt;
143 * </pre>
144 * <pre>
145 * //stop constant check
146 * public static final int someConstant; // won't warn here
147 * //resume constant check
148 * public static final int someConstant; // will warn here as constant's name doesn't match the
149 * // pattern "^[A-Z][A-Z0-9]*$"
150 * </pre>
151 * <p>
152 * To configure a filter so that {@code UNUSED OFF: <i>var</i>} and
153 * {@code UNUSED ON: <i>var</i>} marks a variable or parameter known not to be
154 * used by the code by matching the variable name in the message:
155 * </p>
156 * <pre>
157 * &lt;module name="SuppressionCommentFilter"&gt;
158 *   &lt;property name="offCommentFormat" value="UNUSED OFF\: (\w+)"/&gt;
159 *   &lt;property name="onCommentFormat" value="UNUSED ON\: (\w+)"/&gt;
160 *   &lt;property name="checkFormat" value="Unused"/&gt;
161 *   &lt;property name="messageFormat" value="^Unused \w+ '$1'.$"/&gt;
162 * &lt;/module&gt;
163 * </pre>
164 * <pre>
165 * private static void foo(int a, int b) // UNUSED OFF: b
166 * {
167 * System.out.println(a);
168 * }
169 *
170 * private static void foo1(int a, int b) // UNUSED ON: b
171 * {
172 * System.out.println(a);
173 * }
174 * </pre>
175 * <p>
176 * To configure a filter so that name of suppressed check mentioned in comment
177 * {@code CSOFF: <i>regexp</i>} and {@code CSON: <i>regexp</i>} mark a matching check:
178 * </p>
179 * <pre>
180 * &lt;module name="SuppressionCommentFilter"&gt;
181 *   &lt;property name="offCommentFormat" value="CSOFF\: ([\w\|]+)"/&gt;
182 *   &lt;property name="onCommentFormat" value="CSON\: ([\w\|]+)"/&gt;
183 *   &lt;property name="checkFormat" value="$1"/&gt;
184 * &lt;/module&gt;
185 * </pre>
186 * <pre>
187 * public static final int lowerCaseConstant; // CSOFF: ConstantNameCheck
188 * public static final int lowerCaseConstant1; // CSON: ConstantNameCheck
189 * </pre>
190 * <p>
191 * To configure a filter to suppress all audit events between a comment containing
192 * {@code CHECKSTYLE_OFF: ALMOST_ALL} and a comment containing
193 * {@code CHECKSTYLE_OFF: ALMOST_ALL} except for the <em>EqualsHashCode</em> check:
194 * </p>
195 * <pre>
196 * &lt;module name="SuppressionCommentFilter"&gt;
197 *   &lt;property name="offCommentFormat" value="CHECKSTYLE_OFF: ALMOST_ALL"/&gt;
198 *   &lt;property name="onCommentFormat" value="CHECKSTYLE_ON: ALMOST_ALL"/&gt;
199 *   &lt;property name="checkFormat" value="^((?!(EqualsHashCode)).)*$"/&gt;
200 * &lt;/module&gt;
201 * </pre>
202 * <pre>
203 * public static final int array []; // CHECKSTYLE_OFF: ALMOST_ALL
204 * private String [] strArray;
205 * private int array1 []; // CHECKSTYLE_ON: ALMOST_ALL
206 * </pre>
207 * <p>
208 * To configure a filter to suppress Check's violation message
209 * <b>which matches specified message in messageFormat</b>
210 * (so suppression will be not only by Check's name, but by message text
211 * additionally, as the same Check could report different by message format violations)
212 * between a comment containing {@code stop} and comment containing {@code resume}:
213 * </p>
214 * <pre>
215 * &lt;module name="SuppressionCommentFilter"&gt;
216 *   &lt;property name="offCommentFormat" value="stop"/&gt;
217 *   &lt;property name="onCommentFormat" value="resume"/&gt;
218 *   &lt;property name="checkFormat" value="IllegalTypeCheck"/&gt;
219 *   &lt;property name="messageFormat"
220 *       value="^Declaring variables, return values or parameters of type 'GregorianCalendar'
221 *         is not allowed.$"/&gt;
222 * &lt;/module&gt;
223 * </pre>
224 * <p>
225 * Code before filter above is applied with Check's audit events:
226 * </p>
227 * <pre>
228 * ...
229 * // Warning below: Declaring variables, return values or parameters of type 'GregorianCalendar'
230 * // is not allowed.
231 * GregorianCalendar calendar;
232 * // Warning below here: Declaring variables, return values or parameters of type 'HashSet'
233 * // is not allowed.
234 * HashSet hashSet;
235 * ...
236 * </pre>
237 * <p>
238 * Code after filter is applied:
239 * </p>
240 * <pre>
241 * ...
242 * //stop
243 * GregorianCalendar calendar; // No warning here as it is suppressed by filter.
244 * HashSet hashSet;
245 * // Warning above here: Declaring variables, return values or parameters of type 'HashSet'
246 * //is not allowed.
247 *
248 * //resume
249 * ...
250 * </pre>
251 * <p>
252 * It is possible to specify an ID of checks, so that it can be leveraged by the
253 * SuppressionCommentFilter to skip validations. The following examples show how
254 * to skip validations near code that is surrounded with {@code // CSOFF &lt;ID&gt; (reason)}
255 * and {@code // CSON &lt;ID&gt;}, where ID is the ID of checks you want to suppress.
256 * </p>
257 * <p>
258 * Examples of Checkstyle checks configuration:
259 * </p>
260 * <pre>
261 * &lt;module name="RegexpSinglelineJava"&gt;
262 *   &lt;property name="id" value="ignore"/&gt;
263 *   &lt;property name="format" value="^.*@Ignore\s*$"/&gt;
264 *   &lt;property name="message" value="@Ignore should have a reason."/&gt;
265 * &lt;/module&gt;
266 *
267 * &lt;module name="RegexpSinglelineJava"&gt;
268 *   &lt;property name="id" value="systemout"/&gt;
269 *   &lt;property name="format" value="^.*System\.(out|err).*$"/&gt;
270 *   &lt;property name="message" value="Don't use System.out/err, use SLF4J instead."/&gt;
271 * &lt;/module&gt;
272 * </pre>
273 * <p>
274 * Example of SuppressionCommentFilter configuration (checkFormat which is set
275 * to '$1' points that ID of the checks is in the first group of offCommentFormat
276 * and onCommentFormat regular expressions):
277 * </p>
278 * <pre>
279 * &lt;module name="SuppressionCommentFilter"&gt;
280 *   &lt;property name="offCommentFormat" value="CSOFF (\w+) \(\w+\)"/&gt;
281 *   &lt;property name="onCommentFormat" value="CSON (\w+)"/&gt;
282 *   &lt;property name="idFormat" value="$1"/&gt;
283 * &lt;/module&gt;
284 * </pre>
285 * <pre>
286 * // CSOFF ignore (test has not been implemented yet)
287 * &#64;Ignore // should NOT fail RegexpSinglelineJava
288 * &#64;Test
289 * public void testMethod() { }
290 * // CSON ignore
291 *
292 * // CSOFF systemout (debug)
293 * public static void foo() {
294 *   System.out.println("Debug info."); // should NOT fail RegexpSinglelineJava
295 * }
296 * // CSON systemout
297 * </pre>
298 * <p>
299 * Example of how to configure the check to suppress more than one checks.
300 * </p>
301 * <pre>
302 * &lt;module name="SuppressionCommentFilter"&gt;
303 *   &lt;property name="offCommentFormat" value="@cs-\: ([\w\|]+)"/&gt;
304 *   &lt;property name="checkFormat" value="$1"/&gt;
305 * &lt;/module&gt;
306 * </pre>
307 * <pre>
308 * // @cs-: ClassDataAbstractionCoupling
309 * // @cs-: MagicNumber
310 * &#64;Service // no violations from ClassDataAbstractionCoupling here
311 * &#64;Transactional
312 * public class UserService {
313 *   private int value = 10022; // no violations from MagicNumber here
314 * }
315 * </pre>
316 *
317 * @since 3.5
318 */

467    @Override
468    protected void finishLocalSetup() {
469        // No code by default

472    @Override
473    public boolean accept(TreeWalkerAuditEvent event) {
474        boolean accepted = true;
475
476        if (event.getLocalizedMessage() != null) {
477            // Lazy update. If the first event for the current file, update file
478            // contents and tag suppressions
479            final FileContents currentContents = event.getFileContents();
480
481            if (getFileContents() != currentContents) {
482                setFileContents(currentContents);
483                tagSuppressions();
484            }
485            final Tag matchTag = findNearestMatch(event);
486            accepted = matchTag == null || matchTag.getTagType() == TagType.ON;
487        }
488        return accepted;
489    }
490
491    /**
492     * Finds the nearest comment text tag that matches an audit event.
493     * The nearest tag is before the line and column of the event.
494     *
495     * @param event the {@code TreeWalkerAuditEvent} to match.
496     * @return The {@code Tag} nearest event.
497     */
498    private Tag findNearestMatch(TreeWalkerAuditEvent event) {
499        Tag result = null;
500        for (Tag tag : tags) {
501            if (tag.getLine() > event.getLine()
502                || tag.getLine() == event.getLine()
503                    && tag.getColumn() > event.getColumn()) {
504                break;
505            }
506            if (tag.isMatch(event)) {
507                result = tag;
508            }
509        }
510        return result;
511    }
512
513    /**
514     * Collects all the suppression tags for all comments into a list and
515     * sorts the list.
516     */
517    private void tagSuppressions() {
518        tags.clear();
519        final FileContents contents = getFileContents();
520        if (checkCPP) {
521            tagSuppressions(contents.getSingleLineComments().values());
522        }
523        if (checkC) {
524            final Collection<List<TextBlock>> cComments = contents
525                    .getBlockComments().values();
526            cComments.forEach(this::tagSuppressions);
527        }
528        Collections.sort(tags);
529    }
530
531    /**
532     * Appends the suppressions in a collection of comments to the full
533     * set of suppression tags.
534     *
535     * @param comments the set of comments.
536     */
537    private void tagSuppressions(Collection<TextBlock> comments) {
538        for (TextBlock comment : comments) {
539            final int startLineNo = comment.getStartLineNo();
540            final String[] text = comment.getText();
541            tagCommentLine(text[0], startLineNo, comment.getStartColNo());
542            for (int i = 1; i < text.length; i++) {
543                tagCommentLine(text[i], startLineNo + i, 0);
544            }
545        }
546    }
547
548    /**
549     * Tags a string if it matches the format for turning
550     * checkstyle reporting on or the format for turning reporting off.
551     *
552     * @param text the string to tag.
553     * @param line the line number of text.
554     * @param column the column number of text.
555     */
556    private void tagCommentLine(String text, int line, int column) {
557        final Matcher offMatcher = offCommentFormat.matcher(text);
558        if (offMatcher.find()) {
559            addTag(offMatcher.group(0), line, column, TagType.OFF);
560        }
561        else {
562            final Matcher onMatcher = onCommentFormat.matcher(text);
563            if (onMatcher.find()) {
564                addTag(onMatcher.group(0), line, column, TagType.ON);
565            }
566        }
567    }

569    /**
570     * Adds a {@code Tag} to the list of all tags.
571     *
572     * @param text the text of the tag.
573     * @param line the line number of the tag.
574     * @param column the column number of the tag.
575     * @param reportingOn {@code true} if the tag turns checkstyle reporting on.
576     */
577    private void addTag(String text, int line, int column, TagType reportingOn) {
578        final Tag tag = new Tag(line, column, text, reportingOn, this);
579        tags.add(tag);
580    }

582    /**
583     * A Tag holds a suppression comment and its location, and determines
584     * whether the suppression turns checkstyle reporting on or off.
585     */
586    private static final class Tag
587        implements Comparable<Tag> {
588
589        /** The text of the tag. */
590        private final String text;
591
592        /** The line number of the tag. */
593        private final int line;
594
595        /** The column number of the tag. */
596        private final int column;
597
598        /** Determines whether the suppression turns checkstyle reporting on. */
599        private final TagType tagType;
600
601        /** The parsed check regexp, expanded for the text of this tag. */
602        private final Pattern tagCheckRegexp;
603
604        /** The parsed message regexp, expanded for the text of this tag. */
605        private final Pattern tagMessageRegexp;
606
607        /** The parsed check ID regexp, expanded for the text of this tag. */
608        private final Pattern tagIdRegexp;
609
610        /**
611         * Constructs a tag.
612         *
613         * @param line the line number.
614         * @param column the column number.
615         * @param text the text of the suppression.
616         * @param tagType {@code ON} if the tag turns checkstyle reporting.
617         * @param filter the {@code SuppressionCommentFilter} with the context
618         * @throws IllegalArgumentException if unable to parse expanded text.
619         */
620        /* package */ Tag(int line, int column, String text, TagType tagType,
621                   SuppressionCommentFilter filter) {
622            this.line = line;
623            this.column = column;
624            this.text = text;
625            this.tagType = tagType;
626
627            final Pattern commentFormat;
628            if (this.tagType == TagType.ON) {
629                commentFormat = filter.onCommentFormat;
630            }
631            else {
632                commentFormat = filter.offCommentFormat;
633            }
634
635            // Expand regexp for check and message
636            // Does not intern Patterns with Utils.getPattern()
637            String format = "";
638            try {
639                format = CommonUtil.fillTemplateWithStringsByRegexp(
640                        filter.checkFormat, text, commentFormat);
641                tagCheckRegexp = Pattern.compile(format);
642
643                if (filter.messageFormat == null) {
644                    tagMessageRegexp = null;
645                }
646                else {
647                    format = CommonUtil.fillTemplateWithStringsByRegexp(
648                            filter.messageFormat, text, commentFormat);
649                    tagMessageRegexp = Pattern.compile(format);
650                }
651
652                if (filter.idFormat == null) {
653                    tagIdRegexp = null;
654                }
655                else {
656                    format = CommonUtil.fillTemplateWithStringsByRegexp(
657                            filter.idFormat, text, commentFormat);
658                    tagIdRegexp = Pattern.compile(format);
659                }
660            }
661            catch (final PatternSyntaxException ex) {
662                throw new IllegalArgumentException(
663                    "unable to parse expanded comment " + format, ex);
664            }
665        }
666
667        /**
668         * Returns line number of the tag in the source file.
669         *
670         * @return the line number of the tag in the source file.
671         */
672        public int getLine() {
673            return line;
674        }

678         * Will be 0 for all lines of multiline comment, except the
679         * first line.
680         *
681         * @return the column number of the tag in the source file.
682         */

685        }

692         */
693        public TagType getTagType() {

697        /**
698         * Compares the position of this tag in the file
699         * with the position of another tag.
700         *
701         * @param object the tag to compare with this one.
702         * @return a negative number if this tag is before the other tag,
703         *     0 if they are at the same position, and a positive number if this
704         *     tag is after the other tag.
705         */
706        @Override
707        public int compareTo(Tag object) {
708            final int result;
709            if (line == object.line) {
710                result = Integer.compare(column, object.column);
711            }
712            else {
713                result = Integer.compare(line, object.line);
714            }
715            return result;
716        }
717
718        /**
719         * Indicates whether some other object is "equal to" this one.
720         * Suppression on enumeration is needed so code stays consistent.
721         *
722         * @noinspection EqualsCalledOnEnumConstant
723         */
724        @Override
725        public boolean equals(Object other) {
726            if (this == other) {
727                return true;
728            }
729            if (other == null || getClass() != other.getClass()) {
730                return false;
731            }
732            final Tag tag = (Tag) other;
733            return Objects.equals(line, tag.line)
734                    && Objects.equals(column, tag.column)
735                    && Objects.equals(tagType, tag.tagType)
736                    && Objects.equals(text, tag.text)
737                    && Objects.equals(tagCheckRegexp, tag.tagCheckRegexp)
738                    && Objects.equals(tagMessageRegexp, tag.tagMessageRegexp)
739                    && Objects.equals(tagIdRegexp, tag.tagIdRegexp);
740        }
741
742        @Override
743        public int hashCode() {
744            return Objects.hash(text, line, column, tagType, tagCheckRegexp, tagMessageRegexp,
745                    tagIdRegexp);
746        }
747
748        /**
749         * Determines whether the source of an audit event
750         * matches the text of this tag.
751         *
752         * @param event the {@code TreeWalkerAuditEvent} to check.
753         * @return true if the source of event matches the text of this tag.
754         */
755        public boolean isMatch(TreeWalkerAuditEvent event) {
756            return isCheckMatch(event) && isIdMatch(event) && isMessageMatch(event);
757        }
758
759        /**
760         * Checks whether {@link TreeWalkerAuditEvent} source name matches the check format.
761         *
762         * @param event {@link TreeWalkerAuditEvent} instance.
763         * @return true if the {@link TreeWalkerAuditEvent} source name matches the check format.
764         */
765        private boolean isCheckMatch(TreeWalkerAuditEvent event) {
766            final Matcher checkMatcher = tagCheckRegexp.matcher(event.getSourceName());
767            return checkMatcher.find();
768        }
769
770        /**
771         * Checks whether the {@link TreeWalkerAuditEvent} module ID matches the ID format.
772         *
773         * @param event {@link TreeWalkerAuditEvent} instance.
774         * @return true if the {@link TreeWalkerAuditEvent} module ID matches the ID format.
775         */
776        private boolean isIdMatch(TreeWalkerAuditEvent event) {
777            boolean match = true;
778            if (tagIdRegexp != null) {
779                if (event.getModuleId() == null) {
780                    match = false;
781                }
782                else {
783                    final Matcher idMatcher = tagIdRegexp.matcher(event.getModuleId());
784                    match = idMatcher.find();
785                }
786            }
787            return match;
788        }
789
790        /**
791         * Checks whether the {@link TreeWalkerAuditEvent} message matches the message format.
792         *
793         * @param event {@link TreeWalkerAuditEvent} instance.
794         * @return true if the {@link TreeWalkerAuditEvent} message matches the message format.
795         */
796        private boolean isMessageMatch(TreeWalkerAuditEvent event) {
797            boolean match = true;
798            if (tagMessageRegexp != null) {
799                final Matcher messageMatcher = tagMessageRegexp.matcher(event.getMessage());
800                match = messageMatcher.find();
801            }
802            return match;
803        }

805        @Override
806        public String toString() {
807            return "Tag[text='" + text + '\''
808                    + ", line=" + line
809                    + ", column=" + column
810                    + ", type=" + tagType
811                    + ", tagCheckRegexp=" + tagCheckRegexp
812                    + ", tagMessageRegexp=" + tagMessageRegexp
813                    + ", tagIdRegexp=" + tagIdRegexp + ']';
814        }
