037/**
038 * <p>
039 * Checks that particular classes or interfaces are never used.
040 * </p>
041 * <p>
042 * Rationale: Helps reduce coupling on concrete classes.
043 * </p>
044 * <p>
045 * For additional restriction of type usage see also:
046 * <a href="https://checkstyle.org/config_coding.html#IllegalInstantiation">
047 * IllegalInstantiation</a>,
048 * <a href="https://checkstyle.org/config_imports.html#IllegalImport">IllegalImport</a>
049 * </p>
050 * <p>
051 * It is possible to set illegal class names via short or
052 * <a href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-6.html#jls-6.7">canonical</a>
053 * name. Specifying illegal type invokes analyzing imports and Check puts violations at
054 * corresponding declarations (of variables, methods or parameters).
055 * This helps to avoid ambiguous cases, e.g.: {@code java.awt.List} was set as
056 * illegal class name, then, code like:
057 * </p>
058 * <pre>
059 * import java.util.List;
060 * ...
061 * List list; //No violation here
062 * </pre>
063 * <p>
064 * will be ok.
065 * </p>
066 * <p>
067 * In most cases it's justified to put following classes to <b>illegalClassNames</b>:
068 * </p>
069 * <ul>
070 * <li>GregorianCalendar</li>
071 * <li>Hashtable</li>
072 * <li>ArrayList</li>
073 * <li>LinkedList</li>
074 * <li>Vector</li>
075 * </ul>
076 * <p>
077 * as methods that are differ from interface methods are rarely used, so in most cases user will
078 * benefit from checking for them.
079 * </p>
080 * <ul>
081 * <li>
082 * Property {@code validateAbstractClassNames} - Control whether to validate abstract class names.
083 * Type is {@code boolean}.
084 * Default value is {@code false}.
085 * </li>
086 * <li>
087 * Property {@code illegalClassNames} - Specify classes that should not be used
088 * as types in variable declarations, return values or parameters.
089 * Type is {@code java.lang.String[]}.
090 * Default value is {@code HashMap, HashSet, LinkedHashMap, LinkedHashSet, TreeMap,
091 * TreeSet, java.util.HashMap, java.util.HashSet, java.util.LinkedHashMap,
092 * java.util.LinkedHashSet, java.util.TreeMap, java.util.TreeSet}.
093 * </li>
094 * <li>
095 * Property {@code legalAbstractClassNames} - Define abstract classes that may be used as types.
096 * Type is {@code java.lang.String[]}.
097 * Default value is {@code {}}.
098 * </li>
099 * <li>
100 * Property {@code ignoredMethodNames} - Specify methods that should not be checked.
101 * Type is {@code java.lang.String[]}.
102 * Default value is {@code getEnvironment, getInitialContext}.
103 * </li>
104 * <li>
105 * Property {@code illegalAbstractClassNameFormat} - Specify RegExp for illegal abstract class
106 * names.
107 * Type is {@code java.util.regex.Pattern}.
108 * Default value is {@code "^(.*[.])?Abstract.*$"}.
109 * </li>
110 * <li>
111 * Property {@code memberModifiers} - Control whether to check only methods and fields with any
112 * of the specified modifiers.
113 * This property does not affect method calls nor method references.
114 * Type is {@code int[]}.
115 * Default value is no tokens.
116 * </li>
117 * <li>
118 * Property {@code tokens} - tokens to check
119 * Type is {@code int[]}.
120 * Default value is:
121 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_FIELD_DEF">
122 * ANNOTATION_FIELD_DEF</a>,
123 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
124 * CLASS_DEF</a>,
125 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
126 * INTERFACE_DEF</a>,
127 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_CALL">
128 * METHOD_CALL</a>,
129 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
130 * METHOD_DEF</a>,
131 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_REF">
132 * METHOD_REF</a>,
133 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#PARAMETER_DEF">
134 * PARAMETER_DEF</a>,
135 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#VARIABLE_DEF">
136 * VARIABLE_DEF</a>,
137 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#PATTERN_VARIABLE_DEF">
138 * PATTERN_VARIABLE_DEF</a>.
139 * </li>
140 * </ul>
141 * <p>
142 * Default Configuration:
143 * </p>
144 * <pre>
145 * &lt;module name=&quot;IllegalType&quot;/&gt;
146 * </pre>
147 * <pre>
148 * public class Test extends TreeSet { // violation
149 *   public &lt;T extends java.util.HashSet&gt; void method() { // violation
150 *
151 *     LinkedHashMap&lt;Integer, String&gt; lhmap =
152 *     new LinkedHashMap&lt;Integer, String&gt;(); // violation
153 *     TreeMap&lt;Integer, String&gt; treemap =
154 *     new TreeMap&lt;Integer, String&gt;(); // violation
155 *     Test t; // OK
156 *     HashMap&lt;String, String&gt; hmap; // violation
157 *     Queue&lt;Integer&gt; intqueue; // OK
158 *
159 *     java.lang.IllegalArgumentException illegalex; // OK
160 *     java.util.TreeSet treeset; // violation
161 *   }
162 *
163 * }
164 * </pre>
165 * <p>
166 * To configure the Check so that particular tokens are checked:
167 * </p>
168 * <pre>
169 * &lt;module name="IllegalType"&gt;
170 *   &lt;property name="tokens" value="METHOD_DEF"/&gt;
171 * &lt;/module&gt;
172 * </pre>
173 * <pre>
174 * public class Test extends TreeSet { // OK
175 *   public &lt;T extends java.util.HashSet&gt; void method() { // violation
176 *     LinkedHashMap&lt;Integer, String&gt; lhmap =
177 *     new LinkedHashMap&lt;Integer, String&gt;(); // OK
178 *
179 *     java.lang.IllegalArgumentException illegalex; // OK
180 *     java.util.TreeSet treeset; // Ok
181 *   }
182 *
183 *   public &lt;T extends java.util.HashSet&gt; void typeParam(T t) {} // violation
184 *
185 *   public void fullName(TreeSet a) {} // OK
186 *
187 * }
188 * </pre>
189 * <p>
190 * To configure the Check so that it ignores function() methods:
191 * </p>
192 * <pre>
193 * &lt;module name=&quot;IllegalType&quot;&gt;
194 *   &lt;property name=&quot;ignoredMethodNames&quot; value=&quot;function&quot;/&gt;
195 * &lt;/module&gt;
196 * </pre>
197 * <pre>
198 * public class Test {
199 *   public HashMap&lt;String, String&gt; function() { // OK
200 *     // code
201 *   }
202 *
203 *   public HashMap&lt;String, String&gt; function1() { // violation
204 *     // code
205 *   }
206 * }
207 * </pre>
208 * <p>
209 * To configure the Check so that it validates abstract class names:
210 * </p>
211 * <pre>
212 *  &lt;module name=&quot;IllegalType&quot;&gt;
213 *    &lt;property name=&quot;validateAbstractClassNames&quot; value=&quot;true&quot;/&gt;
214 *    &lt;property name=&quot;illegalAbstractClassNameFormat&quot; value=&quot;Gitt&quot;/&gt;
215 *  &lt;/module&gt;
216 * </pre>
217 * <pre>
218 * class Test extends Gitter { // violation
219 * }
220 *
221 * class Test1 extends Github { // OK
222 * }
223 * </pre>
224 * <p>
225 * To configure the Check so that it verifies only public, protected or static methods and fields:
226 * </p>
227 * <pre>
228 * &lt;module name=&quot;IllegalType&quot;&gt;
229 *   &lt;property name=&quot;memberModifiers&quot; value=&quot;LITERAL_PUBLIC,
230 *    LITERAL_PROTECTED, LITERAL_STATIC&quot;/&gt;
231 * &lt;/module&gt;
232 * </pre>
233 * <pre>
234 * public class Test {
235 *   public HashMap&lt;String, String&gt; function1() { // violation
236 *     // code
237 *   }
238 *
239 *   private HashMap&lt;String, String&gt; function2() { // OK
240 *     // code
241 *   }
242 *
243 *   protected HashMap&lt;Integer, String&gt; function3() { // violation
244 *     // code
245 *   }
246 *
247 *   public static TreeMap&lt;Integer, String&gt; function4() { // violation
248 *     // code
249 *   }
250 *
251 * }
252 * </pre>
253 * <p>
254 * To configure the check so that it verifies usage of types Boolean and Foo:
255 * </p>
256 * <pre>
257 * &lt;module name=&quot;IllegalType&quot;&gt;
258 *           &lt;property name=&quot;illegalClassNames&quot; value=&quot;Boolean, Foo&quot;/&gt;
259 * &lt;/module&gt;
260 * </pre>
261 * <pre>
262 * public class Test {
263 *
264 *   public Set&lt;Boolean&gt; set; // violation
265 *   public java.util.List&lt;Map&lt;Boolean, Foo&gt;&gt; list; // violation
266 *
267 *   private void method(List&lt;Foo&gt; list, Boolean value) { // violation
268 *     SomeType.&lt;Boolean&gt;foo(); // violation
269 *     final Consumer&lt;Foo&gt; consumer = Foo&lt;Boolean&gt;::foo; // violation
270 *   }
271 *
272 *   public &lt;T extends Boolean, U extends Serializable&gt; void typeParam(T a) {} // violation
273 *
274 *   public void fullName(java.util.ArrayList&lt;? super Boolean&gt; a) {} // violation
275 *
276 *   public abstract Set&lt;Boolean&gt; shortName(Set&lt;? super Boolean&gt; a); // violation
277 *
278 *   public Set&lt;? extends Foo&gt; typeArgument() { // violation
279 *     return new TreeSet&lt;Foo&lt;Boolean&gt;&gt;();
280 *   }
281 *
282 * }
283 * </pre>
284 * <p>
285 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
286 * </p>
287 * <p>
288 * Violation Message Keys:
289 * </p>
290 * <ul>
291 * <li>
292 * {@code illegal.type}
293 * </li>
294 * </ul>
295 *
296 * @since 3.2
297 *
298 */
299@FileStatefulCheck
300public final class IllegalTypeCheck extends AbstractCheck {

303     * A key is pointing to the warning message text in "messages.properties"
304     * file.

308    /** Types illegal by default. */
309    private static final String[] DEFAULT_ILLEGAL_TYPES = {
310        "HashSet",
311        "HashMap",
312        "LinkedHashMap",
313        "LinkedHashSet",
314        "TreeSet",
315        "TreeMap",
316        "java.util.HashSet",
317        "java.util.HashMap",
318        "java.util.LinkedHashMap",
319        "java.util.LinkedHashSet",
320        "java.util.TreeSet",
321        "java.util.TreeMap",

325    private static final String[] DEFAULT_IGNORED_METHOD_NAMES = {
326        "getInitialContext",
327        "getEnvironment",

330    /**
331     * Specify classes that should not be used as types in variable declarations,
332     * return values or parameters.
333     */
334    private final Set<String> illegalClassNames = new HashSet<>();
335    /** Illegal short classes. */
336    private final Set<String> illegalShortClassNames = new HashSet<>();
337    /** Define abstract classes that may be used as types. */
338    private final Set<String> legalAbstractClassNames = new HashSet<>();
339    /** Specify methods that should not be checked. */
340    private final Set<String> ignoredMethodNames = new HashSet<>();
341    /**
342     * Control whether to check only methods and fields with any of the specified modifiers.
343     * This property does not affect method calls nor method references.
344     */
345    private List<Integer> memberModifiers;
346
347    /** Specify RegExp for illegal abstract class names. */
348    private Pattern illegalAbstractClassNameFormat = Pattern.compile("^(.*[.])?Abstract.*$");

350    /**
351     * Control whether to validate abstract class names.
352     */
353    private boolean validateAbstractClassNames;

381        return getAcceptableTokens();

384    @Override
385    public int[] getAcceptableTokens() {
386        return new int[] {
387            TokenTypes.ANNOTATION_FIELD_DEF,
388            TokenTypes.CLASS_DEF,
389            TokenTypes.IMPORT,
390            TokenTypes.INTERFACE_DEF,
391            TokenTypes.METHOD_CALL,
392            TokenTypes.METHOD_DEF,
393            TokenTypes.METHOD_REF,
394            TokenTypes.PARAMETER_DEF,
395            TokenTypes.VARIABLE_DEF,
396            TokenTypes.PATTERN_VARIABLE_DEF,
397        };

400    @Override
401    public void beginTree(DetailAST rootAST) {
402        illegalShortClassNames.clear();
403
404        for (String s : illegalClassNames) {
405            if (s.indexOf('.') == -1) {
406                illegalShortClassNames.add(s);
407            }
408        }
409    }
410
411    @Override
412    public int[] getRequiredTokens() {
413        return new int[] {TokenTypes.IMPORT};
414    }
415
416    @Override
417    public void visitToken(DetailAST ast) {
418        switch (ast.getType()) {
419            case TokenTypes.CLASS_DEF:
420            case TokenTypes.INTERFACE_DEF:
421                visitTypeDef(ast);
422                break;
423            case TokenTypes.METHOD_CALL:
424            case TokenTypes.METHOD_REF:
425                visitMethodCallOrRef(ast);
426                break;
427            case TokenTypes.METHOD_DEF:
428                visitMethodDef(ast);
429                break;
430            case TokenTypes.VARIABLE_DEF:
431            case TokenTypes.ANNOTATION_FIELD_DEF:
432            case TokenTypes.PATTERN_VARIABLE_DEF:
433                visitVariableDef(ast);
434                break;
435            case TokenTypes.PARAMETER_DEF:
436                visitParameterDef(ast);
437                break;
438            case TokenTypes.IMPORT:
439                visitImport(ast);
440                break;
441            default:
442                throw new IllegalStateException(ast.toString());
443        }
444    }
445
446    /**
447     * Checks if current method's return type or variable's type is verifiable
448     * according to <b>memberModifiers</b> option.
449     *
450     * @param methodOrVariableDef METHOD_DEF or VARIABLE_DEF ast node.
451     * @return true if member is verifiable according to <b>memberModifiers</b> option.
452     */
453    private boolean isVerifiable(DetailAST methodOrVariableDef) {
454        boolean result = true;
455        if (memberModifiers != null) {
456            final DetailAST modifiersAst = methodOrVariableDef
457                    .findFirstToken(TokenTypes.MODIFIERS);
458            result = isContainVerifiableType(modifiersAst);
459        }
460        return result;
461    }
462
463    /**
464     * Checks is modifiers contain verifiable type.
465     *
466     * @param modifiers
467     *            parent node for all modifiers
468     * @return true if method or variable can be verified
469     */
470    private boolean isContainVerifiableType(DetailAST modifiers) {
471        boolean result = false;
472        if (modifiers.getFirstChild() != null) {
473            for (DetailAST modifier = modifiers.getFirstChild(); modifier != null;
474                     modifier = modifier.getNextSibling()) {
475                if (memberModifiers.contains(modifier.getType())) {
476                    result = true;
477                    break;
478                }
479            }
480        }
481        return result;
482    }
483
484    /**
485     * Checks the super type and implemented interfaces of a given type.
486     *
487     * @param typeDef class or interface for check.
488     */
489    private void visitTypeDef(DetailAST typeDef) {
490        if (isVerifiable(typeDef)) {
491            checkTypeParameters(typeDef);
492            final DetailAST extendsClause = typeDef.findFirstToken(TokenTypes.EXTENDS_CLAUSE);
493            if (extendsClause != null) {
494                checkBaseTypes(extendsClause);
495            }
496            final DetailAST implementsClause = typeDef.findFirstToken(TokenTypes.IMPLEMENTS_CLAUSE);
497            if (implementsClause != null) {
498                checkBaseTypes(implementsClause);
499            }
500        }
501    }
502
503    /**
504     * Checks return type of a given method.
505     *
506     * @param methodDef method for check.
507     */
508    private void visitMethodDef(DetailAST methodDef) {
509        if (isCheckedMethod(methodDef)) {
510            checkClassName(methodDef);
511        }
512    }
513
514    /**
515     * Checks type of parameters.
516     *
517     * @param parameterDef parameter list for check.
518     */
519    private void visitParameterDef(DetailAST parameterDef) {
520        final DetailAST grandParentAST = parameterDef.getParent().getParent();
521
522        if (grandParentAST.getType() == TokenTypes.METHOD_DEF && isCheckedMethod(grandParentAST)) {
523            checkClassName(parameterDef);
524        }
525    }
526
527    /**
528     * Checks type of given variable.
529     *
530     * @param variableDef variable to check.
531     */
532    private void visitVariableDef(DetailAST variableDef) {
533        if (isVerifiable(variableDef)) {
534            checkClassName(variableDef);
535        }
536    }
537
538    /**
539     * Checks the type arguments of given method call/reference.
540     *
541     * @param methodCallOrRef method call/reference to check.
542     */
543    private void visitMethodCallOrRef(DetailAST methodCallOrRef) {
544        checkTypeArguments(methodCallOrRef);
545    }
546
547    /**
548     * Checks imported type (as static and star imports are not supported by Check,
549     *  only type is in the consideration).<br>
550     * If this type is illegal due to Check's options - puts violation on it.
551     *
552     * @param importAst {@link TokenTypes#IMPORT Import}
553     */
554    private void visitImport(DetailAST importAst) {
555        if (!isStarImport(importAst)) {
556            final String canonicalName = getImportedTypeCanonicalName(importAst);
557            extendIllegalClassNamesWithShortName(canonicalName);
558        }
559    }
560
561    /**
562     * Checks if current import is star import. E.g.:
563     * <p>
564     * {@code
565     * import java.util.*;
566     * }
567     * </p>
568     *
569     * @param importAst {@link TokenTypes#IMPORT Import}
570     * @return true if it is star import
571     */
572    private static boolean isStarImport(DetailAST importAst) {
573        boolean result = false;
574        DetailAST toVisit = importAst;
575        while (toVisit != null) {
576            toVisit = getNextSubTreeNode(toVisit, importAst);
577            if (toVisit != null && toVisit.getType() == TokenTypes.STAR) {
578                result = true;
579                break;
580            }
581        }
582        return result;
583    }
584
585    /**
586     * Checks type and type arguments/parameters of given method, parameter, variable or
587     * method call/reference.
588     *
589     * @param ast node to check.
590     */
591    private void checkClassName(DetailAST ast) {
592        final DetailAST type = ast.findFirstToken(TokenTypes.TYPE);
593        checkType(type);
594        checkTypeParameters(ast);
595    }
596
597    /**
598     * Checks the identifier of the given type.
599     *
600     * @param type node to check.
601     */
602    private void checkIdent(DetailAST type) {
603        final FullIdent ident = FullIdent.createFullIdent(type);
604        if (isMatchingClassName(ident.getText())) {
605            log(ident.getDetailAst(), MSG_KEY, ident.getText());
606        }
607    }
608
609    /**
610     * Checks the {@code extends} or {@code implements} statement.
611     *
612     * @param clause DetailAST for either {@link TokenTypes#EXTENDS_CLAUSE} or
613     *               {@link TokenTypes#IMPLEMENTS_CLAUSE}
614     */
615    private void checkBaseTypes(DetailAST clause) {
616        DetailAST child = clause.getFirstChild();
617        while (child != null) {
618            if (child.getType() == TokenTypes.IDENT) {
619                checkIdent(child);
620            }
621            else if (child.getType() == TokenTypes.TYPE_ARGUMENTS) {
622                TokenUtil.forEachChild(child, TokenTypes.TYPE_ARGUMENT, this::checkType);
623            }
624            child = child.getNextSibling();
625        }
626    }
627
628    /**
629     * Checks the given type, its arguments and parameters.
630     *
631     * @param type node to check.
632     */
633    private void checkType(DetailAST type) {
634        checkIdent(type.getFirstChild());
635        checkTypeArguments(type);
636        checkTypeBounds(type);
637    }
638
639    /**
640     * Checks the upper and lower bounds for the given type.
641     *
642     * @param type node to check.
643     */
644    private void checkTypeBounds(DetailAST type) {
645        final DetailAST upperBounds = type.findFirstToken(TokenTypes.TYPE_UPPER_BOUNDS);
646        if (upperBounds != null) {
647            checkType(upperBounds);
648        }
649        final DetailAST lowerBounds = type.findFirstToken(TokenTypes.TYPE_LOWER_BOUNDS);
650        if (lowerBounds != null) {
651            checkType(lowerBounds);
652        }
653    }
654
655    /**
656     * Checks the type parameters of the node.
657     *
658     * @param node node to check.
659     */
660    private void checkTypeParameters(final DetailAST node) {
661        final DetailAST typeParameters = node.findFirstToken(TokenTypes.TYPE_PARAMETERS);
662        if (typeParameters != null) {
663            TokenUtil.forEachChild(typeParameters, TokenTypes.TYPE_PARAMETER, this::checkType);
664        }
665    }
666
667    /**
668     * Checks the type arguments of the node.
669     *
670     * @param node node to check.
671     */
672    private void checkTypeArguments(final DetailAST node) {
673        DetailAST typeArguments = node.findFirstToken(TokenTypes.TYPE_ARGUMENTS);
674        if (typeArguments == null) {
675            typeArguments = node.getFirstChild().findFirstToken(TokenTypes.TYPE_ARGUMENTS);
676        }
677
678        if (typeArguments != null) {
679            TokenUtil.forEachChild(typeArguments, TokenTypes.TYPE_ARGUMENT, this::checkType);
680        }
681    }
682
683    /**
684     * Returns true if given class name is one of illegal classes or else false.
685     *
686     * @param className class name to check.
687     * @return true if given class name is one of illegal classes
688     *         or if it matches to abstract class names pattern.
689     */
690    private boolean isMatchingClassName(String className) {
691        final String shortName = className.substring(className.lastIndexOf('.') + 1);
692        return illegalClassNames.contains(className)
693                || illegalShortClassNames.contains(shortName)
694                || validateAbstractClassNames
695                    && !legalAbstractClassNames.contains(className)
696                    && illegalAbstractClassNameFormat.matcher(className).find();
697    }

699    /**
700     * Extends illegal class names set via imported short type name.
701     *
702     * @param canonicalName
703     *  <a href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-6.html#jls-6.7">
704     *  Canonical</a> name of imported type.
705     */
706    private void extendIllegalClassNamesWithShortName(String canonicalName) {
707        if (illegalClassNames.contains(canonicalName)) {
708            final String shortName = canonicalName
709                .substring(canonicalName.lastIndexOf('.') + 1);
710            illegalShortClassNames.add(shortName);
711        }
712    }

714    /**
715     * Gets imported type's
716     * <a href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-6.html#jls-6.7">
717     *  canonical name</a>.
718     *
719     * @param importAst {@link TokenTypes#IMPORT Import}
720     * @return Imported canonical type's name.
721     */
722    private static String getImportedTypeCanonicalName(DetailAST importAst) {
723        final StringBuilder canonicalNameBuilder = new StringBuilder(256);
724        DetailAST toVisit = importAst;
725        while (toVisit != null) {
726            toVisit = getNextSubTreeNode(toVisit, importAst);
727            if (toVisit != null && toVisit.getType() == TokenTypes.IDENT) {
728                if (canonicalNameBuilder.length() > 0) {
729                    canonicalNameBuilder.append('.');
730                }
731                canonicalNameBuilder.append(toVisit.getText());
732            }
733        }
734        return canonicalNameBuilder.toString();
735    }

737    /**
738     * Gets the next node of a syntactical tree (child of a current node or
739     * sibling of a current node, or sibling of a parent of a current node).
740     *
741     * @param currentNodeAst Current node in considering
742     * @param subTreeRootAst SubTree root
743     * @return Current node after bypassing, if current node reached the root of a subtree
744     *        method returns null
745     */
746    private static DetailAST
747        getNextSubTreeNode(DetailAST currentNodeAst, DetailAST subTreeRootAst) {
748        DetailAST currentNode = currentNodeAst;
749        DetailAST toVisitAst = currentNode.getFirstChild();
750        while (toVisitAst == null) {
751            toVisitAst = currentNode.getNextSibling();
752            if (toVisitAst == null) {
753                if (currentNode.getParent().equals(subTreeRootAst)) {
754                    break;
755                }
756                currentNode = currentNode.getParent();
757            }
758        }
759        return toVisitAst;
760    }

762    /**
763     * Returns true if method has to be checked or false.
764     *
765     * @param ast method def to check.
766     * @return true if we should check this method.
767     */
768    private boolean isCheckedMethod(DetailAST ast) {
769        final String methodName =
770            ast.findFirstToken(TokenTypes.IDENT).getText();
771        return isVerifiable(ast) && !ignoredMethodNames.contains(methodName)
772                && !AnnotationUtil.containsAnnotation(ast, "Override");
773    }
